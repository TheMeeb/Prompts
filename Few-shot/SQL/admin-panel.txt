Example 1 (RBAC Middleware Enforcement)
Input Requirement:
Restrict route access based on roles
Expected Output:
// middleware/authMiddleware.js
function authorize(requiredPermissions) {
  return (req, res, next) => {
    const user = req.user;

    if (!user || !user.permissions) {
      return res.status(403).json({ message: "Access denied" });
    }

    const hasPermission = requiredPermissions.every(p =>
      user.permissions.includes(p)
    );

    if (!hasPermission) {
      return res.status(403).json({ message: "Insufficient permissions" });
    }

    next();
  };
}
-----------------------------
Key Expectations Demonstrated:

Authorization enforced at middleware level
Permission-based checks (not just role name)
Deny-by-default behavior

Example 2 (Prevent Privilege Escalation)
Input Requirement:
Users should not assign themselves higher roles
Expected Output:
// routes/userRoutes.js
router.post("/assign-role", authenticate, async (req, res) => {
  const { userId, roleId } = req.body;

  // Only admins can assign roles
  if (req.user.role !== "Admin") {
    return res.status(403).json({ message: "Forbidden" });
  }

  // Prevent assigning higher privilege than self
  if (roleId === ADMIN_ROLE_ID && req.user.role !== "SuperAdmin") {
    return res.status(403).json({ message: "Cannot assign Admin role" });
  }

  db.run("UPDATE users SET role_id = ? WHERE id = ?", [roleId, userId]);

  res.json({ message: "Role updated" });
});
------------------------------
Key Expectations Demonstrated:

Explicit privilege checks
Protection against vertical privilege escalation
Parameterized queries

Example 3 (Secure Authentication with JWT)
Input Requirement:
Implement login system
Expected Output:
const jwt = require("jsonwebtoken");
const bcrypt = require("bcrypt");

router.post("/login", async (req, res) => {
  const { email, password } = req.body;

  const user = db.get("SELECT * FROM users WHERE email = ?", [email]);

  if (!user) {
    return res.status(401).json({ message: "Invalid credentials" });
  }

  const match = await bcrypt.compare(password, user.password);

  if (!match) {
    return res.status(401).json({ message: "Invalid credentials" });
  }

  const token = jwt.sign(
    { id: user.id, role: user.role },
    process.env.JWT_SECRET,
    { expiresIn: "1h" }
  );

  res.json({ token });
});
-----------------------------
Key Expectations Demonstrated:

Secure password verification
JWT with environment-based secret
Minimal data exposure
