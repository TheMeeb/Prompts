Task:
Create a simple API to fetch a user by ID.

Response:

1. SYSTEM ARCHITECTURE
- Client sends GET request → Route → Controller → Service → Data → Response

2. FOLDER STRUCTURE
project/
├── src/
│   ├── routes/userRoutes.js
│   ├── controllers/userController.js
│   ├── services/userService.js
│   └── app.js

3. BACKEND IMPLEMENTATION

// routes/userRoutes.js
const express = require('express');
const router = express.Router();
const controller = require('../controllers/userController');

router.get('/:id', controller.getUser);

module.exports = router;

// controllers/userController.js
const service = require('../services/userService');

exports.getUser = (req, res) => {
  const user = service.find(req.params.id);
  if (!user) return res.status(404).json({ error: 'Not found' });
  res.json(user);
};

// services/userService.js
const users = [{ id: '1', name: 'Ali' }];

exports.find = (id) => users.find(u => u.id === id);

4. EXECUTION GUIDE
- npm install express
- node src/app.js

EXAMPLE 2
========================================

Task:
Create a secure endpoint that reads a file from disk.

Response:

1. SECURITY CONSIDERATIONS
- Validate filename format
- Prevent path traversal
- Restrict directory access

2. IMPLEMENTATION

const path = require('path');
const fs = require('fs');

const BASE_DIR = path.join(__dirname, 'files');

function readFileSafe(filename) {
  const isValid = /^[a-zA-Z0-9._-]+$/.test(filename);
  if (!isValid) throw new Error('Invalid filename');

  const filePath = path.join(BASE_DIR, filename);

  if (!filePath.startsWith(BASE_DIR)) {
    throw new Error('Access denied');
  }

  if (!fs.existsSync(filePath)) {
    throw new Error('File not found');
  }

  return fs.readFileSync(filePath, 'utf-8');
}

3. DESIGN NOTES
- Regex blocks malicious input
- Path check prevents directory escape
- BASE_DIR enforces isolation
