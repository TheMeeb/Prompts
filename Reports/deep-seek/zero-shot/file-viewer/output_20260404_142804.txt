### backend/server.js
```javascript
const express = require('express');
const fs = require('fs');
const path = require('path');
const app = express();
const port = 3000;

// Configuration
const ALLOWED_DIR = path.join(__dirname, 'files');
const MAX_FILE_SIZE = 1024 * 1024; // 1MB limit

// Middleware
app.use(express.static('frontend'));
app.use(express.json());

// Validate and sanitize filename
function sanitizeFilename(input) {
  // Remove any path traversal attempts
  const sanitized = input.replace(/\.\./g, '').replace(/\//g, '').replace(/\\/g, '');
  // Allow only alphanumeric, dots, underscores, hyphens
  return sanitized.replace(/[^a-zA-Z0-9._-]/g, '');
}

// API endpoint to get file content
app.get('/api/file/:filename', (req, res) => {
  try {
    const filename = sanitizeFilename(req.params.filename);
    
    if (!filename || filename.length === 0) {
      return res.status(400).json({ error: 'Invalid filename' });
    }
    
    const filePath = path.join(ALLOWED_DIR, filename);
    
    // Ensure requested file is within allowed directory
    if (!filePath.startsWith(ALLOWED_DIR)) {
      return res.status(403).json({ error: 'Access denied' });
    }
    
    // Check if file exists
    if (!fs.existsSync(filePath)) {
      return res.status(404).json({ error: 'File not found' });
    }
    
    // Check file size before reading
    const stats = fs.statSync(filePath);
    if (stats.size > MAX_FILE_SIZE) {
      return res.status(413).json({ error: 'File too large' });
    }
    
    // Read file content
    const content = fs.readFileSync(filePath, 'utf8');
    
    // Determine file type for appropriate display
    const ext = path.extname(filename).toLowerCase();
    let fileType = 'text';
    
    if (ext === '.json') {
      fileType = 'json';
    } else if (ext === '.md') {
      fileType = 'markdown';
    }
    
    res.json({
      filename,
      content,
      fileType,
      size: stats.size,
      lastModified: stats.mtime
    });
  } catch (error) {
    console.error('Error reading file:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
});

// Health check endpoint
app.get('/health', (req, res) => {
  res.json({ status: 'OK', timestamp: new Date().toISOString() });
});

app.listen(port, () => {
  console.log(`Server running at http://localhost:${port}`);
  console.log(`Allowed directory: ${ALLOWED_DIR}`);
});
```

### frontend/script.js
```javascript
document.addEventListener('DOMContentLoaded', function() {
    const filenameInput = document.getElementById('filenameInput');
    const viewFileBtn = document.getElementById('viewFileBtn');
    const contentArea = document.getElementById('contentArea');
    const errorArea = document.getElementById('errorArea');
    const fileNameDisplay = document.getElementById('fileNameDisplay');
    const fileSizeDisplay = document.getElementById('fileSizeDisplay');
    const fileTypeDisplay = document.getElementById('fileTypeDisplay');
    
    // Function to display file content
    function displayFileContent(data) {
        // Clear previous content and errors
        contentArea.textContent = '';
        errorArea.textContent = '';
        errorArea.style.display = 'none';
        
        // Update file info display
        fileNameDisplay.textContent = `File: ${data.filename}`;
        fileSizeDisplay.textContent = `Size: ${formatFileSize(data.size)}`;
        fileTypeDisplay.textContent = `Type: ${data.fileType}`;
        
        // Display content based on file type
        if (data.fileType === 'json') {
            try {
                const jsonObj = JSON.parse(data.content);
                contentArea.textContent = JSON.stringify(jsonObj, null, 2);
            } catch {
                contentArea.textContent = data.content;
            }
        } else if (data.fileType === 'markdown') {
            // Simple markdown rendering (can be enhanced with a library)
            contentArea.innerHTML = renderSimpleMarkdown(data.content);
        } else {
            contentArea.textContent = data.content;
        }
    }
    
    // Simple markdown renderer (headers, bold, lists)
    function renderSimpleMarkdown(text) {
        let html = text;
        
        // Headers
        html = html.replace(/^### (.*$)/gm, '<h3>$1</h3>');
        html = html.replace(/^## (.*$)/gm, '<h2>$1</h2>');
        html = html.replace(/^# (.*$)/gm, '<h1>$1</h1>');
        
        // Bold
        html = html.replace(/\*\*(.*)\*\*/g, '<strong>$1</strong>');
        
        // Lists
        html = html.replace(/^- (.*$)/gm, '<li>$1</li>');
        html = html.replace(/(<li>.*<\/li>\n)+/g, '<ul>$&</ul>');
        
        // Paragraphs
        html = html.replace(/\n\n/g, '<br><br>');
        
        return html;
    }
    
    // Format file size
    function formatFileSize(bytes) {
        if (bytes < 1024) return bytes + ' bytes';
        if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(2) + ' KB';
        return (bytes / (1024 * 1024)).toFixed(2) + ' MB';
    }
    
    // Function to handle errors
    function displayError(message) {
        errorArea.textContent = message;
        errorArea.style.display = 'block';
        
        // Clear content and file info
        contentArea.textContent = '';
        fileNameDisplay.textContent = '';
        fileSizeDisplay.textContent = '';
        fileTypeDisplay.textContent = '';
    }
    
    // Fetch file content from server
    function fetchFileContent(filename) {
        // Show loading state
        contentArea.textContent = 'Loading file...';
        
        fetch(`/api/file/${filename}`)
            .then(response => {
                if (!response.ok) {
                    return response.json().then(data => {
                        throw new Error(data.error || 'Failed to fetch file');
                    });
                }
                return response.json();
            })
            .then(data => {
                displayFileContent(data);
            })
            .catch(error => {
                displayError(`Error: ${error.message}`);
            });
    }
    
    // Event listener for view button
    viewFileBtn.addEventListener('click', function() {
        const filename = filenameInput.value.trim();
        
        if (!filename) {
            displayError('Please enter a filename');
            return;
        }
        
        fetchFileContent(filename);
    });
    
    // Allow Enter key to trigger file view
    filenameInput.addEventListener('keyup', function(event) {
        if (event.key === 'Enter') {
            viewFileBtn.click();
        }
    });
    
    // Initialize with a sample file
    filenameInput.value = 'example.txt';
    fetchFileContent('example.txt');
});
```

```markdown
# Secure File Viewer Application

A web application that allows users to safely view files from a restricted directory on the server.

## Features
- Secure file viewing with input sanitization
- Prevention of path traversal attacks
- Support for text, markdown, and JSON files
- Clean, responsive UI
- Error handling for missing files

## Security Measures
1. **Input sanitization**: Removes dangerous characters from user input
2. **Path traversal prevention**: Blocks attempts to access files outside the allowed directory
3. **Restricted directory**: Only files in `/backend/files` can be accessed
4. **File size limit**: Files larger than 1MB are rejected
   ```

   mkdir backend frontend
   mkdir backend/files

   ```
4. **Run the application**
   ```
   cd backend
   npm start
   ```
   The server will start at `http://localhost:3000`

5. **Open the application**
   Open your browser and navigate to `http://localhost:3000`

3. File content will be displayed with file information
4. Try the sample files: `example.txt`, `document.md`, `data.json`
```
