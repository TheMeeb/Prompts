### 3. Database Configuration (src/config/database.js)

```javascript
const sqlite3 = require('sqlite3').verbose();
const path = require('path');

class Database {
  constructor() {
    this.db = new sqlite3.Database(
      path.join(__dirname, '../../blog.db'),
      sqlite3.OPEN_READWRITE | sqlite3.OPEN_CREATE,
      (err) => {
        if (err) {
          console.error('Error opening database:', err.message);
        } else {
          console.log('Connected to SQLite database');
          this.enableForeignKeys();
        }
      }
    );
  }

  enableForeignKeys() {
    this.db.run('PRAGMA foreign_keys = ON');
  }

  query(sql, params = []) {
    return new Promise((resolve, reject) => {
      this.db.all(sql, params, (err, rows) => {
        if (err) {
          reject(err);
        } else {
          resolve(rows);
        }
      });
    });
  }

  run(sql, params = []) {
    return new Promise((resolve, reject) => {
      this.db.run(sql, params, function(err) {
        if (err) {
          reject(err);
        } else {
          resolve({ id: this.lastID, changes: this.changes });
        }
      });
    });
  }

  get(sql, params = []) {
    return new Promise((resolve, reject) => {
      this.db.get(sql, params, (err, row) => {
        if (err) {
          reject(err);
        } else {
          resolve(row);
        }
      });
    });
  }
}

module.exports = new Database();
```

### 4. XSS Sanitization Utility (src/utils/sanitize.js)

```javascript
const xss = require('xss');

const xssOptions = {
  whiteList: {}, // Empty object means no tags are allowed
  stripIgnoreTag: true, // Strip all tags not in whitelist
  stripIgnoreTagBody: ['script', 'style'], // Strip these tags and their content
};

// Custom sanitizer for different input types
const sanitizers = {
  // For titles - allow basic formatting only
  title: (input) => {
    return xss(input, {
      ...xssOptions,
      whiteList: {
        b: [],
        i: [],
        u: [],
        strong: [],
        em: [],
      }
    }).trim().slice(0, 200); // Limit to 200 chars
  },

  // For content - allow more formatting
  content: (input) => {
    return xss(input, {
      ...xssOptions,
      whiteList: {
        p: [],
        br: [],
        b: [],
        i: [],
        u: [],
        strong: [],
        em: [],
        ul: [],
        ol: [],
        li: [],
        h1: [],
        h2: [],
        h3: [],
        h4: [],
        h5: [],
        h6: [],
        blockquote: [],
        code: [],
        pre: [],
      }
    }).trim().slice(0, 10000); // Limit to 10k chars
  },

  // For author names - very restrictive
  authorName: (input) => {
    return xss(input, {
      ...xssOptions,
      whiteList: {
        b: [],
        i: [],
      }
    }).trim().slice(0, 50); // Limit to 50 chars
  },

  // For comments
  comment: (input) => {
    return xss(input, {
      ...xssOptions,
      whiteList: {
        p: [],
        br: [],
        b: [],
        i: [],
        strong: [],
        em: [],
        code: [],
      }
    }).trim().slice(0, 1000); // Limit to 1k chars
  },

  // For general text input
  text: (input) => {
    return xss(input, xssOptions).trim();
  }
};

module.exports = sanitizers;
```

```javascript
const errorHandler = (err, req, res, next) => {
  console.error('Error:', err.stack);

  // Handle validation errors
  if (err.name === 'ValidationError' || Array.isArray(err.array)) {
    return res.status(400).json({
      error: 'Validation Error',
      details: err.array ? err.array() : err.errors
    });
  }

  // Handle database errors
  if (err.code && err.code.startsWith('SQLITE_')) {
    return res.status(500).json({
      error: 'Database Error',
      message: 'An internal error occurred'
    });
  }

  // Handle rate limit errors
  if (err.status === 429) {
    return res.status(429).json({
      error: 'Too Many Requests',
      message: 'Please try again later'
    });
  }

  // Default error
  res.status(err.status || 500).json({
    error: 'Internal Server Error',
    message: process.env.NODE_ENV === 'development' ? err.message : 'Something went wrong'
  });
};

module.exports = errorHandler;
```

```javascript
const inputSanitizer = (req, res, next) => {
  // Sanitize all string properties in body
  if (req.body) {
    for (const key in req.body) {
      if (typeof req.body[key] === 'string') {
        req.body[key] = sanitizers.text(req.body[key]);
      }
    }
  }

  // Sanitize query parameters
  if (req.query) {
    for (const key in req.query) {
      if (typeof req.query[key] === 'string') {
        req.query[key] = sanitizers.text(req.query[key]);
      }
    }
  }

  next();
```

```javascript
const db = require('../config/database');

class Post {
  static async findAll(page = 1, limit = 10) {
    const offset = (page - 1) * limit;
    
    const posts = await db.query(`
      SELECT id, title, content, 
             datetime(created_at, 'localtime') as created_at
      FROM posts 
      ORDER BY created_at DESC 
      LIMIT ? OFFSET ?
    `, [limit, offset]);

    const countResult = await db.get('SELECT COUNT(*) as total FROM posts');
    
    return {
      posts,
      pagination: {
        page: parseInt(page),
        limit: parseInt(limit),
        total: countResult.total,
        pages: Math.ceil(countResult.total / limit)
      }
    };
  }

  static async findById(id) {
    const post = await db.get(`
      SELECT id, title, content, 
             datetime(created_at, 'localtime') as created_at
      FROM posts 
      WHERE id = ?
    `, [id]);
    
    if (!post) {
      throw new Error('Post not found');
    }
    
    return post;
  }

  static async create(title, content) {
    const result = await db.run(
      'INSERT INTO posts (title, content) VALUES (?, ?)',
      [title, content]
    );
    
    return { id: result.id, title, content };
  }

  static async update(id, title, content) {
    const result = await db.run(
      'UPDATE posts SET title = ?, content = ? WHERE id = ?',
      [title, content, id]
    );
    
    if (result.changes === 0) {
      throw new Error('Post not found');
    }
    
    return { id, title, content };
  }

  static async delete(id) {
    const result = await db.run(
      'DELETE FROM posts WHERE id = ?',
      [id]
    );
    
    if (result.changes === 0) {
      throw new Error('Post not found');
    }
    
    return true;
  }
}

module.exports = Post;
```

### 5. API Client (public/js/api.js)

```javascript
const API_BASE_URL = 'http://localhost:3000/api';

class ApiClient {
    constructor() {
        this.defaultHeaders = {
            'Content-Type': 'application/json',
            'Accept': 'application/json'
        };
    }

    async request(endpoint, options = {}) {
        const url = `${API_BASE_URL}${endpoint}`;
        
        const config = {
            ...options,
            headers: {
                ...this.defaultHeaders,
                ...options.headers
            }
        };

        try {
            const response = await fetch(url, config);
            
            if (!response.ok) {
                const errorData = await response.json().catch(() => ({}));
                throw {
                    status: response.status,
                    message: errorData.error || 'Request failed',
                    details: errorData.details
                };
            }

            return await response.json();
        } catch (error) {
            console.error('API request failed:', error);
            throw error;
        }
    }

    // Posts API
    async getPosts(page = 1, limit = 10) {
        return this.request(`/posts?page=${page}&limit=${limit}`);
    }

    async getPost(id) {
        return this.request(`/posts/${id}`);
    }

    async createPost(title, content) {
        return this.request('/posts', {
            method: 'POST',
            body: JSON.stringify({ title, content })
        });
    }

    // Comments API
    async getComments(postId, page = 1, limit = 50) {
        return this.request(`/posts/${postId}/comments?page=${page}&limit=${limit}`);
    }

    async createComment(postId, authorName, content) {
        return this.request(`/posts/${postId}/comments`, {
            method: 'POST',
            body: JSON.stringify({ author_name: authorName, content })
        });
    }
}

// Create singleton instance
const api = new ApiClient();
```

### 6. Main Page JavaScript (public/js/main.js)

```javascript
class BlogApp {
    constructor() {
        this.currentPage = 1;
        this.postsPerPage = 10;
        this.initializeEventListeners();
        this.loadPosts();
        this.setupCharacterCount();
    }

    initializeEventListeners() {
        // New post form
        const newPostForm = document.getElementById('newPostForm');
        if (newPostForm) {
            newPostForm.addEventListener('submit', (e) => this.handleNewPost(e));
        }

        // Character count for textarea
        const postContent = document.getElementById('postContent');
        if (postContent) {
            postContent.addEventListener('input', (e) => this.updateCharCount(e.target, 'charCount'));
        }
    }

    setupCharacterCount() {
        const textareas = document.querySelectorAll('textarea[maxlength]');
        textareas.forEach(textarea => {
            const counterId = textarea.id === 'postContent' ? 'charCount' : 
                             textarea.id === 'commentContent' ? 'commentCharCount' : null;
            if (counterId) {
                textarea.addEventListener('input', (e) => this.updateCharCount(e.target, counterId));
            }
        });
    }

    updateCharCount(textarea, counterId) {
        const counter = document.getElementById(counterId);
        if (counter) {
            counter.textContent = textarea.value.length;
            
            // Visual warning for nearing limit
            const maxLength = parseInt(textarea.getAttribute('maxlength'));
            if (textarea.value.length > maxLength * 0.9) {
                counter.style.color = '#e53e3e';
                counter.style.fontWeight = 'bold';
            } else {
                counter.style.color = '#718096';
                counter.style.fontWeight = 'normal';
            }
        }
    }

    async loadPosts() {
        const loadingElement = document.getElementById('postsLoading');
        const postsContainer = document.getElementById('postsContainer');
        const paginationElement = document.getElementById('pagination');

        try {
            loadingElement.style.display = 'block';
            postsContainer.innerHTML = '';
            paginationElement.innerHTML = '';

            const data = await api.getPosts(this.currentPage, this.postsPerPage);
            
            if (data.posts.length === 0) {
                postsContainer.innerHTML = `
                    <div class="no-posts">
                        <i class="fas fa-newspaper fa-3x"></i>
                        <h3>No posts yet</h3>
                        <p>Be the first to create a post!</p>
                    </div>
                `;
                return;
            }

            // Render posts
            data.posts.forEach(post => {
                const postElement = this.createPostElement(post);
                postsContainer.appendChild(postElement);
            });

            // Render pagination
            if (data.pagination.pages > 1) {
                this.renderPagination(data.pagination);
            }

        } catch (error) {
            console.error('Error loading posts:', error);
            postsContainer.innerHTML = `
                <div class="error-message">
                    <i class="fas fa-exclamation-triangle"></i>
                    <h3>Error loading posts</h3>
                    <p>${error.message || 'Please try again later'}</p>
                </div>
            `;
        } finally {
            loadingElement.style.display = 'none';
        }
    }

    createPostElement(post) {
        const div = document.createElement('div');
        div.className = 'post-card';
        
        // Create excerpt (first 200 characters)
        const excerpt = post.content.length > 200 ? 
            post.content.substring(0, 200) + '...' : 
            post.content;

        div.innerHTML = `
            <h3>${this.escapeHtml(post.title)}</h3>
            <div class="post-date">
                <i class="far fa-calendar"></i>
                ${new Date(post.created_at).toLocaleDateString('en-US', {
                    year: 'numeric',
                    month: 'long',
                    day: 'numeric',
                    hour: '2-digit',
                    minute: '2-digit'
                })}
            </div>
            <div class="post-excerpt">${this.escapeHtml(excerpt)}</div>
            <a href="post.html?id=${post.id}" class="read-more">
                Read More <i class="fas fa-arrow-right"></i>
            </a>
        `;

        return div;
    }

    renderPagination(pagination) {
        const container = document.getElementById('pagination');
        const { page, pages } = pagination;

        let html = '';

        // Previous button
        html += `
            <button class="page-btn" ${page === 1 ? 'disabled' : ''} 
                    onclick="blogApp.changePage(${page - 1})">
                <i class="fas fa-chevron-left"></i> Previous
            </button>
        `;

        // Page numbers
        html += `<span class="current-page">Page ${page} of ${pages}</span>`;

        // Next button
        html += `
            <button class="page-btn" ${page === pages ? 'disabled' : ''} 
                    onclick="blogApp.changePage(${page + 1})">
                Next <i class="fas fa-chevron-right"></i>
            </button>
        `;

        container.innerHTML = html;
    }

    changePage(newPage) {
        if (newPage > 0) {
            this.currentPage = newPage;
            this.loadPosts();
        }
    }

    async handleNewPost(event) {
        event.preventDefault();
        
        const form = event.target;
        const titleInput = document.getElementById('postTitle');
        const contentInput = document.getElementById('postContent');
        const titleError = document.getElementById('titleError');
        const contentError = document.getElementById('contentError');
        const formMessage = document.getElementById('formMessage');

        // Reset errors
        titleError.style.display = 'none';
        contentError.style.display = 'none';
        formMessage.innerHTML = '';

        // Client-side validation
        const title = titleInput.value.trim();
        const content = contentInput.value.trim();
        let isValid = true;

        if (title.length < 3 || title.length > 200) {
            titleError.textContent = 'Title must be between 3 and 200 characters';
            titleError.style.display = 'block';
            isValid = false;
        }

        if (content.length < 10 || content.length > 10000) {
            contentError.textContent = 'Content must be between 10 and 10000 characters';
            contentError.style.display = 'block';
            isValid = false;
        }

        if (!isValid) return;

        try {
            const submitBtn = form.querySelector('button[type="submit"]');
            const originalText = submitBtn.innerHTML;
            submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Publishing...';
            submitBtn.disabled = true;

            const response = await api.createPost(title, content);
            
            formMessage.innerHTML = `
                <div class="success-message">
                    <i class="fas fa-check-circle"></i> Post published successfully!
                </div>
            `;

            form.reset();
            document.getElementById('charCount').textContent = '0';
            
            // Reload posts to show the new one
            setTimeout(() => this.loadPosts(), 1500);

        } catch (error) {
            formMessage.innerHTML = `
                <div class="error-message active">
                    <i class="fas fa-exclamation-triangle"></i>
                    ${error.message || 'Failed to create post'}
                </div>
            `;
        } finally {
            const submitBtn = form.querySelector('button[type="submit"]');
            submitBtn.innerHTML = originalText;
            submitBtn.disabled = false;
        }
    }

    // XSS Prevention: Always escape HTML before rendering
    escapeHtml(text) {
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }

    // Sanitize user input (additional layer)
    sanitizeInput(input) {
        return input
            .replace(/&/g, '&amp;')
            .replace(/</g, '&lt;')
            .replace(/>/g, '&gt;')
            .replace(/"/g, '&quot;')
            .replace(/'/g, '&#x27;');
    }
}

// Initialize app
const blogApp = new BlogApp();
```

```javascript
        const counter = document.getElementById(counterId);
        if (counter) {
            counter.textContent = textarea.value.length;
            
            const maxLength = parseInt(textarea.getAttribute('maxlength'));
            if (textarea.value.length > maxLength * 0.9) {
                counter.style.color = '#e53e3e';
                counter.style.fontWeight = 'bold';
            }
        }
    }

    async loadPost() {
        const postContainer = document.getElementById('postContainer');
        
        try {
            const post = await api.getPost(this.postId);
            this.renderPost(post);
        } catch (error) {
            console.error('Error loading post:', error);
            postContainer.innerHTML = `
                <div class="error-message">
                    <i class="fas fa-exclamation-triangle"></i>
                    <h3>Error loading post</h3>
                    <p>${error.message || 'Post not found'}</p>
                    <a href="index.html" class="btn btn-secondary">
                        <i class="fas fa-arrow-left"></i> Back to Posts
                    </a>
                </div>
            `;
        }
    }

    renderPost(post) {
        const container = document.getElementById('postContainer');
        
        container.innerHTML = `
            <article>
                <h1 class="post-title">${this.escapeHtml(post.title)}</h1>
                <div class="post-meta">
                    <i class="far fa-calendar"></i>
                    ${new Date(post.created_at).toLocaleDateString('en-US', {
                        weekday: 'long',
                        year: 'numeric',
                        month: 'long',
                        day: 'numeric',
                        hour: '2-digit',
                        minute: '2-digit'
                    })}
                </div>
                <div class="post-content">${this.safeHtml(post.content)}</div>
            </article>
        `;
    }

    async loadComments() {
        const loadingElement = document.getElementById('commentsLoading');
        const container = document.getElementById('commentsContainer');
        const paginationElement = document.getElementById('commentsPagination');

        try {
            loadingElement.style.display = 'block';
            container.innerHTML = '';

            const data = await api.getComments(this.postId, this.currentCommentPage, this.commentsPerPage);
            
            if (data.comments.length === 0) {
                container.innerHTML = `
                    <div class="no-comments">
                        <i class="far fa-comments fa-3x"></i>
                        <h3>No comments yet</h3>
                        <p>Be the first to share your thoughts!</p>
                    </div>
                `;
                return;
            }

            // Render comments
            data.comments.forEach(comment => {
                const commentElement = this.createCommentElement(comment);
                container.appendChild(commentElement);
            });

            // Render pagination if needed
            if (data.pagination.pages > 1) {
                this.renderCommentPagination(data.pagination);
            }

        } catch (error) {
            console.error('Error loading comments:', error);
            container.innerHTML = `
                <div class="error-message">
                    <i class="fas fa-exclamation-triangle"></i>
                    <h3>Error loading comments</h3>
                    <p>${error.message || 'Please try again later'}</p>
                </div>
            `;
        } finally {
            loadingElement.style.display = 'none';
        }
    }

    createCommentElement(comment) {
        const div = document.createElement('div');
        div.className = 'comment-card';
        
        div.innerHTML = `
            <div class="comment-header">
                <div class="comment-author">${this.escapeHtml(comment.author_name)}</div>
                <div class="comment-date">
                    <i class="far fa-clock"></i>
                    ${new Date(comment.created_at).toLocaleDateString('en-US', {
                        year: 'numeric',
                        month: 'short',
                        day: 'numeric',
                        hour: '2-digit',
                        minute: '2-digit'
                    })}
                </div>
            </div>
            <div class="comment-content">${this.safeHtml(comment.content)}</div>
        `;

        return div;
    }

    renderCommentPagination(pagination) {
        const container = document.getElementById('commentsPagination');
        const { page, pages } = pagination;

        let html = '';

        if (page > 1) {
            html += `
                <button class="page-btn" onclick="postPage.changeCommentPage(${page - 1})">
                    <i class="fas fa-chevron-left"></i> Older
                </button>
            `;
        }

        html += `<span class="current-page">Page ${page} of ${pages}</span>`;

        if (page < pages) {
            html += `
                <button class="page-btn" onclick="postPage.changeCommentPage(${page + 1})">
                    Newer <i class="fas fa-chevron-right"></i>
                </button>
            `;
        }

        container.innerHTML = html;
    }

    changeCommentPage(newPage) {
        if (newPage > 0) {
            this.currentCommentPage = newPage;
            this.loadComments();
        }
    }

    async handleNewComment(event) {
        event.preventDefault();
        
        const form = event.target;
        const nameInput = document.getElementById('authorName');
        const contentInput = document.getElementById('commentContent');
        const nameError = document.getElementById('nameError');
        const commentError = document.getElementById('commentError');
        const messageElement = document.getElementById('commentMessage');

        // Reset errors and messages
        nameError.style.display = 'none';
        commentError.style.display = 'none';
        messageElement.innerHTML = '';

        // Client-side validation
        const authorName = nameInput.value.trim();
        const content = contentInput.value.trim();
        let isValid = true;

        // Validate name
        if (authorName.length < 2 || authorName.length > 50) {
            nameError.textContent = 'Name must be between 2 and 50 characters';
            nameError.style.display = 'block';
            isValid = false;
        }

        // Validate name format
        const nameRegex = /^[a-zA-Z0-9\s\-_.@]+$/;
        if (!nameRegex.test(authorName)) {
            nameError.textContent = 'Name contains invalid characters. Only letters, numbers, spaces, and basic punctuation are allowed.';
            nameError.style.display = 'block';
            isValid = false;
        }

        // Validate comment
        if (content.length < 1 || content.length > 1000) {
            commentError.textContent = 'Comment must be between 1 and 1000 characters';
            commentError.style.display = 'block';
            isValid = false;
        }

        if (!isValid) return;

        try {
            const submitBtn = form.querySelector('button[type="submit"]');
            const originalText = submitBtn.innerHTML;
            submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Submitting...';
            submitBtn.disabled = true;

            const response = await api.createComment(this.postId, authorName, content);
            
            // Show success message
            messageElement.innerHTML = `
                <div class="success-message">
                    <i class="fas fa-check-circle"></i> Comment added successfully!
                </div>
            `;

            // Reset form
            form.reset();
            document.getElementById('commentCharCount').textContent = '0';
            
            // Reload comments after a delay
            setTimeout(() => {
                this.loadComments();
                // Clear success message after 3 seconds
                setTimeout(() => {
                    messageElement.innerHTML = '';
                }, 3000);
            }, 1000);

        } catch (error) {
            console.error('Error adding comment:', error);
            
            // Handle rate limiting
            if (error.status === 429) {
                messageElement.innerHTML = `
                    <div class="error-message active">
                        <i class="fas fa-hourglass-half"></i>
                        You're submitting comments too quickly. Please wait a few minutes before trying again.
                    </div>
                `;
            } else {
                messageElement.innerHTML = `
                    <div class="error-message active">
                        <i class="fas fa-exclamation-triangle"></i>
                        ${error.message || 'Failed to add comment'}
                    </div>
                `;
            }
        } finally {
            const submitBtn = form.querySelector('button[type="submit']');
            if (submitBtn) {
                submitBtn.innerHTML = originalText;
                submitBtn.disabled = false;
            }
        }
    }

    showError(message) {
        const container = document.querySelector('main');
        container.innerHTML = `
            <div class="error-message">
                <i class="fas fa-exclamation-triangle"></i>
                <h3>Error</h3>
                <p>${message}</p>
                <a href="index.html" class="btn btn-secondary">
                    <i class="fas fa-arrow-left"></i> Back to Posts
                </a>
            </div>
        `;
    }

    // XSS Prevention methods
    escapeHtml(text) {
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }

    // Safe HTML rendering for pre-sanitized content
    safeHtml(content) {
        // Trust the backend sanitization, but still escape to be safe
        return content
            .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, '')
            .replace(/javascript:/gi, '')
            .replace(/on\w+="[^"]*"/gi, '')
            .replace(/on\w+='[^']*'/gi, '')
            .replace(/on\w+=\w+/gi, '');
    }
}

// Initialize post page
const postPage = new PostPage();
```

## Security Implementation Details

### 1. **XSS Prevention (Multi-Layered Defense)**

**Backend Protection:**
- **Input Sanitization**: All user inputs are sanitized using the `xss` library with custom whitelists
- **Strict Whitelisting**: Only safe HTML tags are allowed (no `<script>`, `<iframe>`, etc.)
- **Content-Specific Rules**: Different sanitization rules for titles, content, author names, and comments
- **Output Encoding**: Database returns raw data; frontend escapes before rendering

**Frontend Protection:**
- **DOM-based Escaping**: `escapeHtml()` function converts `<`, `>`, `&`, etc. to HTML entities
- **Text Content Rendering**: Use `textContent` instead of `innerHTML` where possible
- **Safe HTML Parsing**: Custom `safeHtml()` function for pre-sanitized content
- **Content Security Policy**: Configured via Helmet to restrict script sources

### 2. **SQL Injection Prevention**
- **Parameterized Queries**: All database queries use `?` placeholders
- **No String Concatenation**: SQL statements never concatenate user input
- **ORM-like Pattern**: Models use parameterized queries exclusively
- **Input Validation**: Before any database operation, inputs are validated

### 3. **Rate Limiting**
- **General Limiter**: 100 requests per 15 minutes per IP
- **Comment-Specific Limiter**: 10 comments per 5 minutes per IP
- **Graceful Error Handling**: Returns 429 status with user-friendly messages

### 4. **Input Validation & Sanitization**
- **Backend Validation**: Express-validator with custom rules
- **Frontend Validation**: Real-time validation with user feedback
- **Size Limits**: Maximum lengths enforced on both client and server
- **Character Whitelisting**: Author names use regex patterns for allowed characters

### 5. **Additional Security Measures**
- **Helmet.js**: Sets secure HTTP headers (CSP, HSTS, etc.)
- **CORS Configuration**: Restricts API access to known origins
- **Payload Size Limits**: Prevents DoS attacks via large payloads
- **Error Handling**: Generic error messages in production
- **HTTPS Ready**: CORS configured for secure origins

## Key Production Considerations

1. **Environment Variables**: Move configuration to environment variables
2. **HTTPS**: Use HTTPS in production with proper certificates
3. **Database Backups**: Implement regular database backups
4. **Monitoring**: Add logging and monitoring (Winston, PM2)
5. **Load Balancing**: Use Nginx or cloud load balancer
6. **CDN**: For static assets in production
7. **WAF**: Consider a Web Application Firewall for additional protection
8. **Database**: Consider PostgreSQL or MySQL for production

This implementation provides a secure, scalable blog platform with comprehensive protection against common web vulnerabilities while maintaining good performance and user experience.
