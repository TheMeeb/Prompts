```js
function sanitizeText(value) {
  if (typeof value !== "string") return "";
  return value.trim();
}

function escapeHtml(str) {
  return String(str)
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#039;");
}

const ROLE_RANK = {
  User: 1,
  Manager: 2,
  Admin: 3
};

function canManageRole(actorRoleName, targetRoleName) {
  const actorRank = ROLE_RANK[actorRoleName] || 0;
  const targetRank = ROLE_RANK[targetRoleName] || 0;
  return actorRank > targetRank;
}

  sanitizeText,
  escapeHtml,
  ROLE_RANK,
  canManageRole
```

## `backend/middleware/authMiddleware.js`

```js
const jwt = require("jsonwebtoken");
const db = require("../db/database");

function authenticate(req, res, next) {
  const authHeader = req.headers.authorization;

  if (!authHeader || !authHeader.startsWith("Bearer ")) {
    return res.status(401).json({ message: "Authentication required" });
  }

  const token = authHeader.split(" ")[1];

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);

    const user = db.prepare(`
      SELECT users.id, users.username, users.email, roles.name AS role
      FROM users
      JOIN roles ON users.role_id = roles.id
      WHERE users.id = ?
    `).get(decoded.id);

    if (!user) {
      return res.status(401).json({ message: "Invalid token user" });
    }

    const permissions = db.prepare(`
      SELECT permissions.name
      FROM users
      JOIN roles ON users.role_id = roles.id
      JOIN role_permissions ON roles.id = role_permissions.role_id
      JOIN permissions ON permissions.id = role_permissions.permission_id
      WHERE users.id = ?
    `).all(user.id).map(row => row.name);

    req.user = {
      id: user.id,
      username: user.username,
      email: user.email,
      role: user.role,
      permissions
    };

    next();
  } catch (err) {
    return res.status(401).json({ message: "Invalid or expired token" });
  }
}

module.exports = authenticate;
```

```js
      message: "Validation failed",
```

```js
  (req, res) => {
    const users = db.prepare(`

router.post(
  "/",
  authenticate,
  authorize(["users.create"]),

    body("password").isLength({ min: 8, max: 128 }),

      message: "User created",
```

```js
const express = require("express");
const { body, param } = require("express-validator");
const db = require("../db/database");
const authenticate = require("../middleware/authMiddleware");
const authorize = require("../middleware/authorizeMiddleware");

router.get(
  "/",
  authenticate,
  authorize(["roles.read"]),
  (req, res) => {
    const roles = db.prepare("SELECT id, name FROM roles ORDER BY id ASC").all();

    const permissionsByRoleStmt = db.prepare(`
      SELECT permissions.name
      FROM role_permissions
      JOIN permissions ON permissions.id = role_permissions.permission_id
      WHERE role_permissions.role_id = ?
      ORDER BY permissions.name
    `);

    const result = roles.map(role => ({
      ...role,
      permissions: permissionsByRoleStmt.all(role.id).map(r => r.name)
    }));

    res.json(result);
  }
);

router.get(
  "/permissions/all",
  authenticate,
  authorize(["roles.read"]),
  (req, res) => {
    const permissions = db.prepare("SELECT id, name FROM permissions ORDER BY name ASC").all();
    res.json(permissions);
  }
);

router.put(
  "/:roleId/permissions",
  authenticate,
  authorize(["roles.update"]),
  [
    param("roleId").isInt({ min: 1 }),
    body("permissionIds").isArray()
  ],
  validate,
  (req, res) => {
    const roleId = Number(req.params.roleId);
    const { permissionIds } = req.body;

    const role = db.prepare("SELECT id, name FROM roles WHERE id = ?").get(roleId);
    if (!role) {
      return res.status(404).json({ message: "Role not found" });
    }

    if (ROLE_RANK[req.user.role] <= ROLE_RANK[role.name]) {
      return res.status(403).json({ message: "Cannot modify equal or higher role permissions" });
    }

    const placeholders = permissionIds.map(() => "?").join(",");
    let validPermissionRows = [];

    if (permissionIds.length > 0) {
      validPermissionRows = db.prepare(
        `SELECT id FROM permissions WHERE id IN (${placeholders})`
      ).all(...permissionIds);

      if (validPermissionRows.length !== permissionIds.length) {
        return res.status(400).json({ message: "Invalid permission IDs" });
      }
    }

    const deleteStmt = db.prepare("DELETE FROM role_permissions WHERE role_id = ?");
    const insertStmt = db.prepare(`
      INSERT INTO role_permissions (role_id, permission_id)
      VALUES (?, ?)
    `);

    const transaction = db.transaction(() => {
      deleteStmt.run(roleId);
      for (const permissionId of permissionIds) {
        insertStmt.run(roleId, permissionId);
      }
    });

    transaction();

    res.json({ message: "Role permissions updated" });
  }
);

module.exports = router;
```

```js
  allowedHeaders: ["Content-Type", "Authorization"]
}));

app.use(express.json({ limit: "10kb" }));

const authLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 20,
  message: { message: "Too many login attempts, try again later." }
});

app.use("/api/auth", authLimiter, authRoutes);
app.use("/api/users", userRoutes);
app.use("/api/roles", roleRoutes);
app.use("/api/permissions", permissionRoutes);

app.get("/api/health", (req, res) => {
  res.json({ status: "ok" });
});

app.use(errorHandler);

const PORT = process.env.PORT || 4000;
app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
```

## `frontend/js/api.js`

```js
const API_BASE = "http://localhost:4000/api";

function getToken() {
  return localStorage.getItem("token");
}

function getUser() {
  const raw = localStorage.getItem("user");
  return raw ? JSON.parse(raw) : null;
}

function logout() {
  localStorage.removeItem("token");
  localStorage.removeItem("user");
  window.location.href = "./index.html";
}

async function apiRequest(path, options = {}) {
  const token = getToken();

  const headers = {
    "Content-Type": "application/json",
    ...(options.headers || {})
  };

  if (token) {
    headers.Authorization = `Bearer ${token}`;
  }

  const response = await fetch(`${API_BASE}${path}`, {
    ...options,
    headers
  });

  const data = await response.json().catch(() => ({}));

  if (response.status === 401) {
    logout();
    throw new Error("Session expired");
  }

  if (!response.ok) {
    throw new Error(data.message || "Request failed");
  }

  return data;
}

function requireAuth() {
  if (!getToken() || !getUser()) {
    window.location.href = "./index.html";
  }
}

function safeText(value) {
  return String(value ?? "");
}
```

Already included above:

- `backend/middleware/authMiddleware.js`
- `backend/middleware/authorizeMiddleware.js`

These enforce:

- JWT verification
- User lookup from DB
- Permission resolution from DB
- Route-level permission checks

# Explanation of How Role & Permission Checks Are Enforced

## 1. Authentication enforcement
Every protected route uses:

```js
authenticate
```

This middleware:

- Reads the `Authorization: Bearer <token>` header
- Verifies JWT using `process.env.JWT_SECRET`
- Loads the current user from the database
- Loads the user’s effective permissions from:
  - `users`
  - `roles`
  - `role_permissions`
  - `permissions`
- Attaches user context to `req.user`

If anything fails, access is denied with `401`.

---

## 2. Authorization enforcement
Protected routes also use:

```js
authorize(["some.permission"])
```

Example:

```js
router.get("/", authenticate, authorize(["users.read"]), ...)
```

This means:

- If the user is not authenticated: denied
- If the user lacks any required permission: denied
- Deny-by-default behavior is enforced

Authorization is based on permissions, not only role names.

---

## 3. Privilege escalation prevention
The app prevents vertical privilege escalation in several ways:

### User creation
A user cannot create another user with an equal or higher role unless explicitly allowed.

```js
if (!canManageRole(req.user.role, targetRole.name) && req.user.role !== targetRole.name)
```

### User updates
A user cannot:

- modify equal/higher privileged users
- assign equal/higher roles
- change their own role

### User deletion
A user cannot:

- delete themselves
- delete equal/higher privileged users

### Role permission updates
A user cannot change permissions for their own role level or higher:

```js
if (ROLE_RANK[req.user.role] <= ROLE_RANK[role.name])
```

This prevents a Manager from granting Admin-level capabilities.

---

## 4. SQL injection prevention
All SQL queries use parameterized statements only.

Examples:

```js
db.prepare("SELECT * FROM users WHERE email = ?").get(email)
db.prepare("DELETE FROM users WHERE id = ?").run(userId)
```

No string concatenation is used for user-controlled values.

---

## 5. Input validation and sanitization
Validation is done with `express-validator`.

Examples:

- email format checks
- integer checks for IDs
- password length checks
- string trimming and escaping

This reduces malformed input and injection risk.

---

## 6. XSS protection
Backend:
- `helmet()` sets secure headers
- inputs are validated and escaped where appropriate

Frontend:
- user data is rendered with `textContent`
- no unsafe `innerHTML` is used for untrusted data

This prevents script injection into the UI.

---

## 7. Sensitive data protection
The API never returns password hashes.

User responses return only:

- id
- username
- email
- role
- permissions where needed

JWT payload contains only:

```js
{ id: user.id }
```

This minimizes exposure.

---

## 8. CSRF
This project uses JWT in the `Authorization` header, not cookies.

Therefore classic browser cookie-based CSRF risk is significantly reduced.

If you later switch to cookie auth, add CSRF tokens and `SameSite` cookies.

# Important Security Notes

- Backend authorization is the real enforcement point.
- Frontend role-based rendering is only for UX.
- Even if a user manually calls an API from the browser console, backend middleware still enforces permissions.
- JWT secrets must come from environment variables.
- Never store plaintext passwords.
- Do not trust role labels from the frontend; backend derives permissions from the DB.
