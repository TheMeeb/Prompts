```js
const { clientOrigin } = require('../config/env');

function corsMiddleware(req, res, next) {
  const origin = req.headers.origin;

  if (!origin || origin === clientOrigin) {
    res.setHeader('Access-Control-Allow-Origin', clientOrigin);
    res.setHeader('Vary', 'Origin');
    res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization');
    res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
    res.setHeader('Access-Control-Allow-Credentials', 'false');

    if (req.method === 'OPTIONS') {
      return res.sendStatus(204);
    }

    return next();
  }

  return res.status(403).json({ error: 'Origin not allowed' });
}

module.exports = { corsMiddleware };
```

### 3.8 `src/middleware/rateLimiter.js`

```js
class RateLimiter {
  constructor({ windowMs, maxRequests }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = new Map();
  }

  isLimited(key) {
    const now = Date.now();
    const entry = this.store.get(key);

    if (!entry) {
      this.store.set(key, { count: 1, start: now });
      return false;
    }

    if (now - entry.start > this.windowMs) {
      this.store.set(key, { count: 1, start: now });
      return false;
    }

    entry.count += 1;

    if (entry.count > this.maxRequests) {
      return true;
    }

    return false;
  }

  cleanup() {
    const now = Date.now();
    for (const [key, value] of this.store.entries()) {
      if (now - value.start > this.windowMs * 2) {
        this.store.delete(key);
      }
    }
  }
}

module.exports = { RateLimiter };
```

```js
function validateChatMessage(payload) {
  if (!payload || typeof payload !== 'object' || Array.isArray(payload)) {
    return { valid: false, error: 'Payload must be an object' };

  if (type !== 'chat_message') {
```

```js
function issueChatToken(username) {
```

```js
const { buildSystemMessage } = require('./messageService');
const { enableSqlite } = require('../config/env');
const db = enableSqlite ? require('../db/sqlite') : null;

class ChatService {
  constructor() {
    this.clients = new Map(); // ws -> { username, room, ip }
  }

  addClient(ws, clientData) {
    this.clients.set(ws, clientData);
  }

  removeClient(ws) {
    const client = this.clients.get(ws);
    this.clients.delete(ws);
    return client;
  }

  getClient(ws) {
    return this.clients.get(ws);
  }

  broadcast(wss, payload, room = 'global') {
    const data = JSON.stringify(payload);

    for (const client of wss.clients) {
      if (client.readyState !== 1) continue;

      const meta = this.clients.get(client);
      if (!meta) continue;
      if ((meta.room || 'global') !== room) continue;

      client.send(data);
    }
  }

  notifyJoin(wss, username, room = 'global') {
    const message = buildSystemMessage(`${username} joined the chat`, room);
    this.broadcast(wss, message, room);
  }

  notifyLeave(wss, username, room = 'global') {
    const message = buildSystemMessage(`${username} left the chat`, room);
    this.broadcast(wss, message, room);
  }

  persistMessage(message) {
    if (db) {
      db.saveMessage(message);
    }
  }

  getRecentMessages(room = 'global', limit = 20) {
    if (!db) return [];
    return db.getRecentMessages(room, limit);
  }
}

module.exports = new ChatService();
```

### 3.15 `src/sockets/socketHandlers.js`

```js
const chatService = require('../services/chatService');
const { verifyChatToken } = require('../services/authService');
const { buildChatMessage, safeJsonParse } = require('../services/messageService');
const { validateChatMessage } = require('../utils/validation');
const { RateLimiter } = require('../middleware/rateLimiter');
const { clientOrigin } = require('../config/env');

const messageRateLimiter = new RateLimiter({ windowMs: 5000, maxRequests: 10 });
setInterval(() => messageRateLimiter.cleanup(), 30000);

function getTokenFromRequest(req) {
  const url = new URL(req.url, 'http://localhost');
  return url.searchParams.get('token');
}

function handleConnection(wss, ws, req) {
  const origin = req.headers.origin;
  if (origin !== clientOrigin) {
    ws.close(1008, 'Unauthorized origin');
    return;
  }

  const token = getTokenFromRequest(req);
  if (!token) {
    ws.close(1008, 'Missing token');
    return;
  }

  let decoded;
  try {
    decoded = verifyChatToken(token);
  } catch {
    ws.close(1008, 'Invalid token');
    return;
  }

  const ip = req.socket.remoteAddress || 'unknown';
  const username = decoded.username;
  const room = 'global';

  chatService.addClient(ws, { username, room, ip });

  const recentMessages = chatService.getRecentMessages(room, 20);
  ws.send(JSON.stringify({
    type: 'history',
    room,
    messages: recentMessages
  }));

  chatService.notifyJoin(wss, username, room);

  ws.on('message', (rawMessage, isBinary) => {
    try {
      if (isBinary) {
        ws.send(JSON.stringify({ type: 'error', error: 'Binary messages are not allowed' }));
        return;
      }

      const rateKey = `${ip}:${username}`;
      if (messageRateLimiter.isLimited(rateKey)) {
        ws.send(JSON.stringify({ type: 'error', error: 'Rate limit exceeded' }));
        return;
      }

      const parsed = safeJsonParse(rawMessage.toString());
      if (!parsed.ok) {
        ws.send(JSON.stringify({ type: 'error', error: 'Malformed JSON' }));
        return;
      }

      const validation = validateChatMessage(parsed.data);
      if (!validation.valid) {
        ws.send(JSON.stringify({ type: 'error', error: validation.error }));
        return;
      }

      const finalMessage = buildChatMessage({
        username,
        message: parsed.data.message,
        room
      });

      chatService.persistMessage(finalMessage);
      chatService.broadcast(wss, finalMessage, room);
    } catch (err) {
      console.error('Socket message handling error:', err);
      ws.send(JSON.stringify({ type: 'error', error: 'Unable to process message' }));
    }
  });

  ws.on('close', () => {
    const client = chatService.removeClient(ws);
    if (client) {
      chatService.notifyLeave(wss, client.username, client.room);
    }
  });

  ws.on('error', (err) => {
    console.error('WebSocket error:', err.message);
  });
}

module.exports = {
  handleConnection
};
```

### 4.3 `public/app.js`

```js
const usernameInput = document.getElementById('username');
const connectBtn = document.getElementById('connectBtn');
const statusEl = document.getElementById('status');
const messagesEl = document.getElementById('messages');
const chatForm = document.getElementById('chatForm');
const messageInput = document.getElementById('messageInput');
const sendBtn = document.getElementById('sendBtn');

let socket = null;
let token = null;

function setStatus(text, connected = false) {
  statusEl.textContent = text;
  statusEl.className = `status ${connected ? 'connected' : 'disconnected'}`;
}

function escapeHtml(input = '') {
  return String(input)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}

function appendMessage({ type, username, message, timestamp }) {
  const wrapper = document.createElement('div');
  wrapper.className = `message ${type === 'system_notification' ? 'system' : ''}`;

  const meta = document.createElement('span');
  meta.className = 'meta';

  const safeUser = escapeHtml(username || 'unknown');
  const safeTime = escapeHtml(new Date(timestamp).toLocaleTimeString());
  meta.innerHTML = `${safeUser} • ${safeTime}`;

  const body = document.createElement('div');
  body.textContent = message;

  wrapper.appendChild(meta);
  wrapper.appendChild(body);
  messagesEl.appendChild(wrapper);
  messagesEl.scrollTop = messagesEl.scrollHeight;
}

async function fetchToken(username) {
  const response = await fetch('/api/auth/token', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({ username })
  });

  if (!response.ok) {
    throw new Error('Failed to get token');
  }

  const data = await response.json();
  return data.token;
}

function connectSocket() {
  const protocol = location.protocol === 'https:' ? 'wss:' : 'ws:';
  socket = new WebSocket(`${protocol}//${location.host}/ws?token=${encodeURIComponent(token)}`);

  socket.addEventListener('open', () => {
    setStatus('Connected', true);
    messageInput.disabled = false;
    sendBtn.disabled = false;
  });

  socket.addEventListener('message', (event) => {
    try {
      const data = JSON.parse(event.data);

      if (data.type === 'history' && Array.isArray(data.messages)) {
        messagesEl.innerHTML = '';
        data.messages.forEach(appendMessage);
        return;
      }

      if (data.type === 'error') {
        appendMessage({
          type: 'system_notification',
          username: 'system',
          message: data.error,
          timestamp: new Date().toISOString()
        });
        return;
      }

      if (data.type === 'chat_message' || data.type === 'system_notification') {
        appendMessage(data);
      }
    } catch {
      appendMessage({
        type: 'system_notification',
        username: 'system',
        message: 'Received invalid server response',
        timestamp: new Date().toISOString()
      });
    }
  });

  socket.addEventListener('close', () => {
    setStatus('Disconnected', false);
    messageInput.disabled = true;
    sendBtn.disabled = true;
  });

  socket.addEventListener('error', () => {
    setStatus('Connection Error', false);
  });
}

connectBtn.addEventListener('click', async () => {
  const username = usernameInput.value.trim();

  if (!username) {
    alert('Username is required');
    return;
  }

  try {
    token = await fetchToken(username);
    connectSocket();
  } catch {
    alert('Unable to connect');
  }
});

chatForm.addEventListener('submit', (event) => {
  event.preventDefault();

  const message = messageInput.value.trim();
  if (!message || !socket || socket.readyState !== WebSocket.OPEN) {
    return;
  }

  socket.send(JSON.stringify({
    type: 'chat_message',
    message
  }));

  messageInput.value = '';
});
```

```js
const Database = require('better-sqlite3');
const fs = require('fs');
const path = require('path');

const dbPath = path.join(__dirname, 'chat.db');
const schemaPath = path.join(__dirname, 'schema.sql');

const db = new Database(dbPath);
const schema = fs.readFileSync(schemaPath, 'utf8');
db.exec(schema);

const insertMessageStmt = db.prepare(`
  INSERT INTO messages (type, username, message, room, timestamp)
  VALUES (@type, @username, @message, @room, @timestamp)
`);

const recentMessagesStmt = db.prepare(`
  SELECT type, username, message, room, timestamp
  FROM messages
  WHERE room = ?
  ORDER BY id DESC
  LIMIT ?
`);

function saveMessage(message) {
  insertMessageStmt.run(message);
}

function getRecentMessages(room, limit = 20) {
  return recentMessagesStmt.all(room, limit).reverse();
}

module.exports = {
  saveMessage,
  getRecentMessages
};
```

## 6. SECURITY ANALYSIS

### 6.1 XSS Prevention
**Server-side**
- All user-controlled text is sanitized with `sanitizeText()`
- Username and message fields are escaped before storage and broadcast
- Join/leave notifications are also sanitized

**Client-side**
- Rendering uses `textContent` for message body
- Any dynamic string inserted in HTML is escaped with `escapeHtml()`
- No raw untrusted HTML is injected into DOM

**Result**
- Even if attacker sends `<script>alert(1)</script>`, it is rendered as plain text

---

### 6.2 Validation Strategy
- Every inbound message is parsed using safe JSON parsing
- Message must be an object
- `type` must equal `chat_message`
- `message` must be a non-empty string under 500 chars
- Optional `room` must be valid if implemented
- Binary messages are rejected
- Username validated before token issuance

**Zero trust model**
- Client validation is convenience only
- Server performs authoritative validation

---

### 6.3 Authentication
- Client first requests a JWT token via `/api/auth/token`
- WebSocket connection requires valid `token` query parameter
- Invalid or expired tokens are rejected with close code `1008`
- Username identity comes from signed token, not from incoming messages

**Production note**
- In a real deployment, replace demo token issuance with:
  - session auth
  - OAuth
  - external identity provider
  - signed HttpOnly cookie-based auth

---

### 6.4 Rate Limiting
- Per-user/IP message limiter
- Example policy: max 10 messages / 5 seconds
- Exceeding threshold returns error and suppresses message processing

**Why**
- Prevents flooding
- Reduces abuse
- Protects server resources

---

### 6.5 Origin/CORS Checks
- HTTP requests restricted by `corsMiddleware`
- WebSocket handshake checks `req.headers.origin`
- Any origin not matching configured frontend origin is rejected

**Important**
- Origin checks help mitigate unauthorized browser-based use
- They do not replace authentication

---

### 6.6 Safe Error Handling
- Invalid JSON does not crash the server
- Bad payloads return structured error messages
- Socket event handlers are wrapped in `try/catch`
- HTTP errors are passed through centralized error middleware
- Sensitive internal stack traces are not exposed to clients

---

### 6.7 Additional Security Best Practices
- `helmet` sets secure headers
- JSON body size limited
- No trust in client username after socket connection
- Message size constrained
- CSP reduces client-side script injection risk
