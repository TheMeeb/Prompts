```js
const crypto = require('crypto');

async function createExecutionDir() {
  const baseDir = path.join(os.tmpdir(), 'online-ide');
  await fs.mkdir(baseDir, { recursive: true });

  const dirName = `exec-${Date.now()}-${crypto.randomUUID()}`;
```

### 3.6 `backend/src/services/docker.service.js`

```js
const { spawn } = require('child_process');

function runDockerContainer({
  image,
  workspace,
  command,
  timeoutMs = 3000,
  memory = '128m',
  cpus = '0.5',
  pidsLimit = '64'
}) {
  return new Promise((resolve) => {
    const args = [
      'run',
      '--rm',
      '--network', 'none',
      '--memory', memory,
      '--cpus', cpus,
      '--pids-limit', pidsLimit,
      '--read-only',
      '--cap-drop', 'ALL',
      '--security-opt', 'no-new-privileges',
      '--tmpfs', '/tmp:rw,noexec,nosuid,size=64m',
      '--user', '1000:1000',
      '-v', `${workspace}:/workspace:ro`,
      image,
      ...command
    ];

    const child = spawn('docker', args, {
      stdio: ['ignore', 'pipe', 'pipe'],
      env: {},
      shell: false
    });

    let stdout = '';
    let stderr = '';
    let finished = false;

    const timer = setTimeout(() => {
      if (!finished) {
        child.kill('SIGKILL');
      }
    }, timeoutMs);

    child.stdout.on('data', (data) => {
      stdout += data.toString();
    });

    child.stderr.on('data', (data) => {
      stderr += data.toString();
    });

    child.on('error', () => {
      clearTimeout(timer);
      finished = true;
      resolve({
        success: false,
        stdout: '',
        stderr: 'Execution service unavailable',
        exitCode: 1
      });
    });

    child.on('close', (code, signal) => {
      clearTimeout(timer);
      finished = true;

      if (signal === 'SIGKILL') {
        return resolve({
          success: false,
          stdout,
          stderr: 'Execution timed out',
          exitCode: 124
        });
      }

      resolve({
        success: code === 0,
        stdout,
        stderr,
        exitCode: code
      });
    });
  });
}

module.exports = {
  runDockerContainer
};
```

### 3.7 `backend/src/services/execution.service.js`

```js
const fs = require('fs/promises');
const path = require('path');
const languages = require('../config/languages');
const { createExecutionDir, removeDir } = require('../utils/tempDir');
const { runDockerContainer } = require('./docker.service');

async function executeCode({ language, code }) {
  const config = languages[language];
  if (!config) {
    throw new Error('Unsupported language');
  }

  const execDir = await createExecutionDir();

  try {
    const filePath = path.join(execDir, config.fileName);
    await fs.writeFile(filePath, code, { encoding: 'utf8', mode: 0o600 });

    const result = await runDockerContainer({
      image: config.image,
      workspace: execDir,
      command: config.runCommand,
      timeoutMs: 3000,
      memory: '128m',
      cpus: '0.5',
      pidsLimit: '64'
    });

    return {
      success: result.success,
      output: result.stdout,
      error: result.stderr,
      exitCode: result.exitCode
    };
  } finally {
    await removeDir(execDir);
  }
}

module.exports = {
  executeCode
};
```

```js
      data.error ? `STDERR:\n${data.error}` : '',
      `EXIT CODE: ${data.exitCode}`
    ].filter(Boolean).join('\n\n');

    outputConsole.textContent = output || 'No output';
  } catch (err) {
    outputConsole.textContent = 'Unable to connect to server';
  } finally {
    runBtn.disabled = false;
  }
});
```

## 5. SANDBOX IMPLEMENTATION

## Docker isolation strategy

This design uses **one container per execution request**.

### Why this is secure
- Code is **not executed on the host**
- Each run is isolated from other users
- The container is destroyed after execution
- The container has:
  - **No network**
  - **Read-only filesystem**
  - **Dropped Linux capabilities**
  - **No privilege escalation**
  - **Restricted memory/CPU/processes**
  - **No inherited environment variables**
  - **Non-root execution user**

### Actual Docker command generated by backend

Conceptually, backend runs:

```bash
docker run --rm \
  --network none \
  --memory 128m \
  --cpus 0.5 \
  --pids-limit 64 \
  --read-only \
  --cap-drop ALL \
  --security-opt no-new-privileges \
  --tmpfs /tmp:rw,noexec,nosuid,size=64m \
  --user 1000:1000 \
  -v /tmp/online-ide/<exec-id>:/workspace:ro \
  node:20-alpine \
  node /workspace/main.js
```

For Python:

```bash
docker run --rm \
  --network none \
  --memory 128m \
  --cpus 0.5 \
  --pids-limit 64 \
  --read-only \
  --cap-drop ALL \
  --security-opt no-new-privileges \
  --tmpfs /tmp:rw,noexec,nosuid,size=64m \
  --user 1000:1000 \
  -v /tmp/online-ide/<exec-id>:/workspace:ro \
  python:3.11-alpine \
  python /workspace/main.py
```

### Isolation details
- `--rm`  
  Removes container after execution.

- `--network none`  
  Blocks outbound/inbound network access.

- `--memory 128m`  
  Prevents memory abuse.

- `--cpus 0.5`  
  Restricts CPU consumption.

- `--pids-limit 64`  
  Reduces fork-bomb/process abuse.

- `--read-only`  
  Prevents writes to container filesystem.

- `--tmpfs /tmp`  
  Provides limited writable temp space only.

- `--cap-drop ALL`  
  Removes Linux capabilities.

- `--security-opt no-new-privileges`  
  Prevents privilege escalation.

- `--user 1000:1000`  
  Avoids root in container.

- `env: {}` in `spawn()`  
  Prevents backend environment leakage.

### Important production note
For higher security in production, consider:
- Rootless Docker
- gVisor / Kata Containers / Firecracker microVMs
- Dedicated isolated worker nodes
- Seccomp/AppArmor profiles
- Queue-based async execution service

## 6. SECURITY ANALYSIS

## 6.1 Command Execution Safety
- Uses `spawn('docker', args, { shell: false })`
- Does **not** use `exec()` with raw shell strings
- User code is written to file and passed to language runtime safely
- User input is never interpolated into a shell command

## 6.2 Sandboxing
- Execution happens inside Docker, not on host
- One execution gets one disposable container
- No shared execution state between requests
- Read-only mount limits file tampering
- No host environment variables passed to subprocess

## 6.3 Resource Limits
- **Timeout:** enforced in Node process
- **Memory:** `128m`
- **CPU:** `0.5`
- **PIDs:** `64`
- Prevents:
  - infinite loops
  - heavy memory usage
  - process spawning abuse

## 6.4 Input Validation
- Only supported languages allowed
- Code must be string
- Code size capped
- Empty payload rejected
- JSON body size limited with `express.json({ limit: '50kb' })`

## 6.5 Rate Limiting
- `express-rate-limit`
- Example: 20 execution requests per minute per IP
- Reduces API abuse and brute-force resource exhaustion

## 6.6 Secure Error Handling
- Internal errors return generic message
- Does not expose stack traces to clients
- Logs only minimal server-side information

## 6.7 Zero Trust Model
All code is treated as malicious:
- no host execution
- no shell execution
- no network
- no writable host access
- no env inheritance
- limited runtime resources

## 6.8 Remaining Risks
Even with Docker, untrusted code execution is sensitive. For production internet-scale use:
- isolate execution workers from API nodes
- use VM/microVM isolation
- scan/maintain runner images
- implement audit logging and queueing
- use stronger kernel hardening and seccomp profiles

- Pushes job to Redis/RabbitMQ/SQS
- Worker service performs execution
- Result stored temporarily
- Client polls or uses WebSocket

Benefits:
- better load control
- horizontal worker scaling
- isolation from main API
