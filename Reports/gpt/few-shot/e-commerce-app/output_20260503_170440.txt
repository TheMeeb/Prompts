```js
const { verifyToken } = require('../utils/jwt');

function authenticate(req, res, next) {
  const authHeader = req.headers.authorization;

  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    return res.status(401).json({ message: 'Authentication required' });
  }

  const token = authHeader.split(' ')[1];

  try {
    const decoded = verifyToken(token);
    req.user = decoded;
    next();
  } catch {
    return res.status(401).json({ message: 'Invalid or expired token' });
  }
}

module.exports = { authenticate };
```

```js
const bcrypt = require('bcrypt');
const db = require('../config/db');
const { signToken } = require('../utils/jwt');

async function registerUser({ username, email, password }) {
  const existing = db.prepare(
    'SELECT id FROM users WHERE email = ? OR username = ?'
  ).get(email, username);

  if (existing) {
    const error = new Error('User already exists');
    error.status = 409;
    throw error;
  }

  const hashedPassword = await bcrypt.hash(password, 12);

  const stmt = db.prepare(`
    INSERT INTO users (username, email, password, role)
    VALUES (?, ?, ?, 'user')
  `);

  const result = stmt.run(username, email, hashedPassword);

  return { id: result.lastInsertRowid, username, email, role: 'user' };
}

async function loginUser({ email, password }) {
  const user = db.prepare(
    'SELECT id, username, email, password, role FROM users WHERE email = ?'
  ).get(email);

  if (!user) {
    const error = new Error('Invalid credentials');
    error.status = 401;
    throw error;
  }

  const match = await bcrypt.compare(password, user.password);

  if (!match) {
    const error = new Error('Invalid credentials');
    error.status = 401;
    throw error;
  }

  const token = signToken({
    id: user.id,
    username: user.username,
    role: user.role
  });

  return {
    token,
    user: {
      id: user.id,
      username: user.username,
      email: user.email,
      role: user.role
    }
  };
}

module.exports = { registerUser, loginUser };
```

```js
const db = require('../config/db');

function getAllProducts(filters = {}) {
  let query = 'SELECT id, name, description, price, stock, category FROM products WHERE 1=1';
  const params = [];

  if (filters.name) {
    query += ' AND name LIKE ?';
    params.push(`%${filters.name}%`);
  }

  if (filters.category) {
    query += ' AND category = ?';
    params.push(filters.category);
  }

  if (filters.minPrice !== undefined) {
    query += ' AND price >= ?';
    params.push(filters.minPrice);
  }

  if (filters.maxPrice !== undefined) {
    query += ' AND price <= ?';
    params.push(filters.maxPrice);
  }

  query += ' ORDER BY id DESC';

  return db.prepare(query).all(...params);
}

function getProductById(id) {
  const product = db.prepare(
    'SELECT id, name, description, price, stock, category FROM products WHERE id = ?'
  ).get(id);

  if (!product) {
    const error = new Error('Product not found');
    error.status = 404;
    throw error;
  }

  return product;
}

function createProduct(data) {
  const stmt = db.prepare(`
    INSERT INTO products (name, description, price, stock, category)
    VALUES (?, ?, ?, ?, ?)
  `);

  const result = stmt.run(
    data.name,
    data.description,
    data.price,
    data.stock,
    data.category || null
  );

  return getProductById(result.lastInsertRowid);
}

function updateProduct(id, data) {
  const existing = db.prepare('SELECT id FROM products WHERE id = ?').get(id);
  if (!existing) {
    const error = new Error('Product not found');
    error.status = 404;
    throw error;
  }

  db.prepare(`
    UPDATE products
    SET name = ?, description = ?, price = ?, stock = ?, category = ?
    WHERE id = ?
  `).run(data.name, data.description, data.price, data.stock, data.category || null, id);

  return getProductById(id);
}

function deleteProduct(id) {
  const result = db.prepare('DELETE FROM products WHERE id = ?').run(id);
  if (result.changes === 0) {
    const error = new Error('Product not found');
    error.status = 404;
    throw error;
  }
}

  getAllProducts,

  createProduct,
  updateProduct,
```

### `src/services/orderService.js`
```js
const db = require('../config/db');

function createOrder(userId, items) {
  const transaction = db.transaction(() => {
    let totalPrice = 0;
    const validatedItems = [];

    for (const item of items) {
      const product = db.prepare(
        'SELECT id, name, price, stock FROM products WHERE id = ?'
      ).get(item.productId);

      if (!product) {
        const error = new Error(`Invalid product: ${item.productId}`);
        error.status = 400;
        throw error;
      }

      if (item.quantity <= 0) {
        const error = new Error('Invalid quantity');
        error.status = 400;
        throw error;
      }

      if (product.stock < item.quantity) {
        const error = new Error(`Insufficient stock for ${product.name}`);
        error.status = 400;
        throw error;
      }

      const lineTotal = product.price * item.quantity;
      totalPrice += lineTotal;

      validatedItems.push({
        productId: product.id,
        quantity: item.quantity,
        price: product.price
      });
    }

    const orderResult = db.prepare(`
      INSERT INTO orders (user_id, total_price, created_at)
      VALUES (?, ?, datetime('now'))
    `).run(userId, totalPrice);

    const orderId = orderResult.lastInsertRowid;

    for (const item of validatedItems) {
      db.prepare(`
        INSERT INTO order_items (order_id, product_id, quantity)
        VALUES (?, ?, ?)
      `).run(orderId, item.productId, item.quantity);

      db.prepare(`
        UPDATE products
        SET stock = stock - ?
        WHERE id = ?
      `).run(item.quantity, item.productId);
    }

    return {
      orderId,
      totalPrice,
      items: validatedItems
    };
  });

  return transaction();
}

function getUserOrders(userId) {
  return db.prepare(`
    SELECT id, total_price, created_at
    FROM orders
    WHERE user_id = ?
    ORDER BY created_at DESC
  `).all(userId);
}

function getAllOrders() {
  return db.prepare(`
    SELECT o.id, o.user_id, u.username, o.total_price, o.created_at
    FROM orders o
    JOIN users u ON o.user_id = u.id
    ORDER BY o.created_at DESC
  `).all();
}

module.exports = { createOrder, getUserOrders, getAllOrders };
```

### `frontend/js/api.js`
```js
const API_BASE = 'http://localhost:5000/api';

function getToken() {
  return localStorage.getItem('token');
}

async function apiFetch(endpoint, options = {}) {
  const headers = {
    'Content-Type': 'application/json',
    ...(options.headers || {})
  };

  const token = getToken();
  if (token) {
    headers.Authorization = `Bearer ${token}`;
  }

  const response = await fetch(`${API_BASE}${endpoint}`, {
    ...options,
    headers
  });

  const data = await response.json();

  if (!response.ok) {
    throw new Error(data.message || 'API request failed');
  }

  return data;
}
```

```js
function escapeHtml(str = '') {
  const div = document.createElement('div');
  div.innerText = str;
  return div.innerHTML;
}

async function loadProducts() {
  const name = document.getElementById('searchName')?.value || '';
  const category = document.getElementById('searchCategory')?.value || '';
  const minPrice = document.getElementById('minPrice')?.value || '';
  const maxPrice = document.getElementById('maxPrice')?.value || '';

  const params = new URLSearchParams();
  if (name) params.append('name', name);
  if (category) params.append('category', category);
  if (minPrice) params.append('minPrice', minPrice);
  if (maxPrice) params.append('maxPrice', maxPrice);

  try {
    const products = await apiFetch(`/products?${params.toString()}`);
    const container = document.getElementById('productList');

    container.innerHTML = products.map(product => `
      <div class="card">
        <h3>${escapeHtml(product.name)}</h3>
        <p>${escapeHtml(product.description)}</p>
        <p><strong>$${product.price.toFixed(2)}</strong></p>
        <p>Stock: ${product.stock}</p>
        <button onclick="addToCart(${product.id})">Add to Cart</button>
      </div>
    `).join('');
  } catch (err) {
    alert(err.message);
  }
}

function addToCart(productId) {
  const cart = JSON.parse(localStorage.getItem('cart') || '[]');
  const existing = cart.find(i => i.productId === productId);

  if (existing) {
    existing.quantity += 1;
  } else {
    cart.push({ productId, quantity: 1 });
  }

  localStorage.setItem('cart', JSON.stringify(cart));
  alert('Added to cart');
}
```

```js
async function loadCart() {
  const cart = JSON.parse(localStorage.getItem('cart') || '[]');
  const container = document.getElementById('cartItems');

  if (cart.length === 0) {
    container.innerHTML = '<p>Your cart is empty</p>';
    return;
  }

  let html = '';

  for (const item of cart) {
    try {
      const product = await apiFetch(`/products/${item.productId}`);
      html += `
        <div class="card">
          <h3>${product.name}</h3>
          <p>Quantity: ${item.quantity}</p>
          <p>Unit Price: $${product.price.toFixed(2)}</p>
          <button onclick="removeFromCart(${item.productId})">Remove</button>
        </div>
      `;
    } catch {
      html += `<div class="card"><p>Product unavailable</p></div>`;
    }
  }

  container.innerHTML = html;
}

function removeFromCart(productId) {
  const cart = JSON.parse(localStorage.getItem('cart') || '[]')
    .filter(item => item.productId !== productId);

  localStorage.setItem('cart', JSON.stringify(cart));
  loadCart();
}
```

# 5. DATABASE SCHEMA

## `src/database/schema.sql`
```sql
PRAGMA foreign_keys = ON;

CREATE TABLE IF NOT EXISTS users (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  username TEXT NOT NULL UNIQUE,
  email TEXT NOT NULL UNIQUE,
  password TEXT NOT NULL,
  role TEXT NOT NULL CHECK(role IN ('user', 'admin'))
);

CREATE TABLE IF NOT EXISTS products (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL,
  description TEXT NOT NULL,
  price REAL NOT NULL CHECK(price >= 0),
  stock INTEGER NOT NULL CHECK(stock >= 0),
  category TEXT
);

CREATE TABLE IF NOT EXISTS orders (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id INTEGER NOT NULL,
  total_price REAL NOT NULL CHECK(total_price >= 0),
  created_at TEXT NOT NULL,
  FOREIGN KEY(user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS order_items (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  order_id INTEGER NOT NULL,
  product_id INTEGER NOT NULL,
  quantity INTEGER NOT NULL CHECK(quantity > 0),
  FOREIGN KEY(order_id) REFERENCES orders(id) ON DELETE CASCADE,
  FOREIGN KEY(product_id) REFERENCES products(id) ON DELETE CASCADE
);

CREATE INDEX IF NOT EXISTS idx_products_name ON products(name);
CREATE INDEX IF NOT EXISTS idx_products_category ON products(category);
CREATE INDEX IF NOT EXISTS idx_orders_user_id ON orders(user_id);
CREATE INDEX IF NOT EXISTS idx_order_items_order_id ON order_items(order_id);
```

# 6. SECURITY ANALYSIS

## 6.1 Authentication & RBAC
- **JWT authentication** used for stateless auth
- Passwords hashed using **bcrypt**
- Admin-only routes protected by:
  - `authenticate`
  - `authorize('admin')`

Example:
```js
router.get('/users', authenticate, authorize('admin'), adminController.getUsers);
```

## 6.2 SQL Injection Prevention
- SQLite queries use **parameterized statements**
- No unsafe string concatenation for user values in SQL parameters

Example:
```js
db.prepare('SELECT id, username, email, password, role FROM users WHERE email = ?').get(email);
```

## 6.3 Input Validation & Sanitization
- `express-validator` validates:
  - email format
  - password length
  - numeric prices
  - quantity constraints
- `.escape()` used on text fields
- Frontend uses DOM escaping before injecting user-controlled content

## 6.4 XSS Prevention
- Frontend sanitizes rendered strings:
```js
function escapeHtml(str = '') {
  const div = document.createElement('div');
  div.innerText = str;
  return div.innerHTML;
}
```
- Backend uses Helmet to set security headers
- Avoids raw HTML rendering from user input

## 6.5 E-Commerce Checkout Security
- Client cart is **not trusted**
- Server recalculates total price from DB
- Server validates:
  - product exists
  - quantity > 0
  - stock availability
- Stock reduced only inside **transaction**
- Prevents price tampering and order manipulation

Example:
```js
const product = db.prepare(
  'SELECT id, name, price, stock FROM products WHERE id = ?'
).get(item.productId);
```

## 6.6 Rate Limiting
- Login/register protected from brute force
- Checkout protected from abuse

## 6.7 Secure Error Handling
- Generic 500 responses
- No stack traces leaked in production
- Centralized error middleware

## 6.8 Environment Security
- JWT secret stored in `.env`
- Secrets not hardcoded in application logic

## 6.9 CSRF Protection
Because this implementation uses **Authorization Bearer tokens in headers**, browser cookies are not used for auth, so classic CSRF risk is significantly reduced.

If switching to cookie-based auth:
- add `csurf`
- use `SameSite=Strict`
- `HttpOnly`
- `Secure` cookies
