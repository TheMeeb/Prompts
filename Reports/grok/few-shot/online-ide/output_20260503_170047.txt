function runCode(filePath) {

    execFile(
      'node',
      [filePath],

**1. SYSTEM ARCHITECTURE**

```
User Browser
    ↓ (HTTPS)
Express API (Node.js)
    ↓ (validated & sanitized)
Execution Service (isolated worker)
```

- All user code treated as malicious
- Execution never happens on host directly
- No shell execution (`exec`, `spawn` with shell)
- Containers are ephemeral and destroyed after each run
- Least privilege + seccomp + AppArmor / SELinux equivalent

```js
const validateExecution = (req, res, next) => {
  const { code, language } = req.body;

  if (typeof code !== 'string' || code.length > 50000) {
    return res.status(400).json({ error: 'Code must be a string under 50KB' });
  }

  const allowed = ['javascript', 'python', 'java'];
  if (!allowed.includes(language)) {
    return res.status(400).json({ error: 'Language not supported' });
  }

  // Basic dangerous pattern detection (defense in depth)
  if (language === 'javascript' && /require\(|process\.|child_process|fs\./.test(code)) {
    return res.status(400).json({ error: 'Use of dangerous APIs is not allowed' });
  }

  next();
};

module.exports = { validateExecution };
```

### `backend/src/services/sandbox.js` (Core Execution Engine)

```js
const { spawn } = require('child_process');
const path = require('path');
const { v4: uuidv4 } = require('uuid');

const SUPPORTED_LANGUAGES = {
  javascript: { image: 'node:18-slim', cmd: ['node', '/sandbox/code.js'] },
  python:     { image: 'python:3.11-slim', cmd: ['python', '/sandbox/code.py'] },
  java:       { image: 'openjdk:17-slim', cmd: ['java', '/sandbox/Main'] }
};

async function executeInSandbox(code, language) {
  const executionId = uuidv4();
  const langConfig = SUPPORTED_LANGUAGES[language];

  if (!langConfig) throw new Error('Unsupported language');

  const args = [
    'run', '--rm',
    '--memory=256m',
    '--cpus=0.5',
    '--network=none',
    '--security-opt=no-new-privileges',
    '--cap-drop=ALL',
    '--read-only',
    '--tmpfs=/tmp:noexec,size=10m',
    '-v', `${path.join('/tmp', executionId)}:/sandbox:ro`,
    langConfig.image,
    ...langConfig.cmd
  ];

  // Write code to temporary directory (cleaned up after)
  const fs = require('fs').promises;
  const tempDir = `/tmp/${executionId}`;
  await fs.mkdir(tempDir, { recursive: true });

  const filename = language === 'javascript' ? 'code.js' :
                   language === 'python' ? 'code.py' : 'Main.java';

  await fs.writeFile(path.join(tempDir, filename), code);

  if (language === 'java') {
    // Simple Java setup (for demo - production would compile properly)
    await fs.writeFile(path.join(tempDir, 'Main.java'), code);
    // In real production: compile step before running container
  }

  return new Promise((resolve) => {
    const docker = spawn('docker', args, { timeout: 8000 });

    let stdout = '';
    let stderr = '';

    docker.stdout.on('data', (data) => { stdout += data; });
    docker.stderr.on('data', (data) => { stderr += data; });

    docker.on('close', async (code) => {
      // Cleanup
      await fs.rm(tempDir, { recursive: true, force: true }).catch(() => {});

      resolve({
        output: stdout.trim(),
        error: stderr.trim(),
        exitCode: code,
        executionId
      });
    });

    docker.on('error', async (err) => {
      await fs.rm(tempDir, { recursive: true, force: true }).catch(() => {});
      resolve({ error: 'Sandbox execution failed', exitCode: 1 });
    });

    // Kill after timeout
    setTimeout(() => {
      docker.kill();
      resolve({ error: 'Execution timeout', exitCode: 124 });
    }, 7000);
  });
}

module.exports = { executeInSandbox };
```

### `backend/src/controllers/executionController.js`
```js
const { executeInSandbox } = require('../services/sandbox');

async function executeCode(req, res) {
  try {
    const { code, language } = req.body;
    const result = await executeInSandbox(code, language);

    // Never expose internal paths or system info
    const sanitized = {
      output: result.output || '',
      error: result.error || '',
      success: result.exitCode === 0
    };

    res.json(sanitized);
  } catch (err) {
    console.error('Execution error:', err.message);
    res.status(500).json({ error: 'Internal execution error' });
  }
}

module.exports = { executeCode };
```

**5. SANDBOX IMPLEMENTATION (Docker)**

**Key Security Features:**

```yaml
# docker-compose.yml (for development)
services:
  api:
    build: ./backend
    ports: ["4000:4000"]
    environment:
      - NODE_ENV=production
    depends_on:
      - docker-socket-proxy

  docker-socket-proxy:
    image: tecnativa/docker-socket-proxy
    restart: always
    volumes:
      - /var/run/docker.sock:/var/run/docker.sock:ro
    environment:
      - CONTAINERS=1
      - IMAGES=0
      - VOLUMES=0
```

**Sandbox Strategy:**
- `docker-socket-proxy` with least-privilege (only `CONTAINERS=1`)
- `--read-only` + `--network=none`
- Memory + CPU limits
- No privileged mode
- Seccomp profile (can be hardened further with custom profile)
- Temporary bind mount (read-only where possible)
- Container is removed immediately after execution (`--rm`)

---

**6. SECURITY ANALYSIS**

| Threat | Mitigation |
|-------|----------|
| Code injection / RCE | Full containerization + no shell |
| Fork bomb / infinite loop | CPU + Memory limits + timeout |
| Filesystem escape | Read-only root + tmpfs with `noexec` |
| Network attacks | `--network=none` |
| Privilege escalation | `--no-new-privileges`, `--cap-drop=ALL` |
| Large payloads | Input size limit + rate limiting |
| Information disclosure | Sanitized error responses |
