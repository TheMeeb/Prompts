### 1. Backend (`server.js`)

```js
const express = require('express');
const fs = require('fs').promises;
const path = require('path');

const app = express();
const PORT = 3000;

// Security: Only allow files from this directory
const ALLOWED_DIR = path.join(__dirname, 'files');

// Middleware
app.use(express.static('public'));
app.use(express.json());

// Security middleware to prevent path traversal
const validateFilename = (filename) => {
    if (!filename || typeof filename !== 'string') {
        throw new Error('Invalid filename');
    }
    
    // Remove any path separators and dangerous characters
    const sanitized = filename.replace(/[/\\]/g, '').trim();
    
    if (sanitized === '' || sanitized === '.' || sanitized === '..') {
        throw new Error('Invalid filename');
    }
    
    return sanitized;
};

app.post('/api/view', async (req, res) => {
    try {
        const { filename } = req.body;
        
        const safeFilename = validateFilename(filename);
        
        // Resolve full path and ensure it's within allowed directory
        const filePath = path.join(ALLOWED_DIR, safeFilename);
        
        // Double-check that the resolved path is still within ALLOWED_DIR
        if (!filePath.startsWith(ALLOWED_DIR + path.sep)) {
            return res.status(403).json({ 
                error: 'Access denied: Path traversal attempt detected' 
            });
        }

        // Check if file exists
        try {
            await fs.access(filePath);
        } catch {
            return res.status(404).json({ 
                error: `File '${safeFilename}' not found` 
            });
        }

        const content = await fs.readFile(filePath, 'utf-8');

        res.json({
            filename: safeFilename,
            content: content,
            success: true
        });

    } catch (error) {
        console.error('Error:', error.message);
        res.status(400).json({
            error: error.message || 'Failed to read file'
        });
    }
});

// Create sample files on first run
async function initializeSampleFiles() {
    try {
        await fs.mkdir(ALLOWED_DIR, { recursive: true });
        
        const samples = {
            'example.txt': 'Hello World!\n\nThis is a sample text file.\nYou can create more files in the /files directory.',
            'notes.md': '# Project Notes\n\n- Remember to validate all user inputs\n- Always prevent path traversal\n- Keep sensitive files outside web root',
            'readme.txt': 'Welcome to the Secure File Viewer!\n\nOnly files inside the "files" directory can be viewed.'
        };

        for (const [filename, content] of Object.entries(samples)) {
            const filePath = path.join(ALLOWED_DIR, filename);
            try {
                await fs.access(filePath);
            } catch {
                await fs.writeFile(filePath, content);
            }
        }
        console.log('✅ Sample files created in /files directory');
    } catch (err) {
        console.error('Failed to create sample files:', err.message);
    }
}

// Start server
async function startServer() {
    await initializeSampleFiles();
    
    app.listen(PORT, () => {
        console.log(`🚀 Server running at http://localhost:${PORT}`);
        console.log(`📁 Serving files from: ${ALLOWED_DIR}`);
    });
}

startServer();
```

#### `public/script.js`

```js
const filenameInput = document.getElementById('filename');
const viewBtn = document.getElementById('viewBtn');
const errorDiv = document.getElementById('error');
const successDiv = document.getElementById('success');
const contentPre = document.getElementById('content');
const fileNameDisplay = document.getElementById('fileNameDisplay');

function showError(message) {
    errorDiv.textContent = message;
    errorDiv.style.display = 'block';
    successDiv.style.display = 'none';
    contentPre.textContent = '';
    fileNameDisplay.textContent = 'Error';
}

function showSuccess(filename, content) {
    errorDiv.style.display = 'none';
    successDiv.textContent = `✅ Successfully loaded: ${filename}`;
    successDiv.style.display = 'block';
    
    fileNameDisplay.textContent = filename;
    contentPre.textContent = content;
}

viewBtn.addEventListener('click', async () => {
    const filename = filenameInput.value.trim();
    
    if (!filename) {
        showError('Please enter a filename');
        return;
    }

    try {
        const response = await fetch('/api/view', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ filename })
        });

        const data = await response.json();

        if (!response.ok) {
            showError(data.error || 'Failed to fetch file');
        } else {
            showSuccess(data.filename, data.content);
        }
    } catch (err) {
        showError('Network error. Please try again.');
        console.error(err);
    }
});

// Allow pressing Enter
filenameInput.addEventListener('keypress', (e) => {
    if (e.key === 'Enter') {
        viewBtn.click();
    }
});

// Bonus: Clear error when user starts typing
filenameInput.addEventListener('input', () => {
    if (errorDiv.style.display === 'block') {
        errorDiv.style.display = 'none';
    }
});
```

### Security Features Implemented

- **Path Traversal Protection**: `../`, absolute paths, and `../../etc/passwd` are blocked
- **Input Sanitization**: Removes slashes and dangerous characters
- **Directory Restriction**: All files must be in `/files` folder
- **Path Resolution Validation**: Uses `path.join()` + `startsWith()` check
- **No System Files Exposure**: Cannot access `package.json`, `.env`, etc.
- **Clear Error Messages**: Doesn't leak file system information
