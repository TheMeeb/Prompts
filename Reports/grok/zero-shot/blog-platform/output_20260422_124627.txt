### Database Schema (`database/schema.sql`)

```sql
-- Schema
CREATE TABLE IF NOT EXISTS posts (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT NOT NULL,
    content TEXT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS comments (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    post_id INTEGER NOT NULL,
    author_name TEXT NOT NULL,
    content TEXT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE
);

-- Indexes for performance
CREATE INDEX IF NOT EXISTS idx_comments_post_id ON comments(post_id);
CREATE INDEX IF NOT EXISTS idx_posts_created_at ON posts(created_at);
```

const sanitizeOptions = {
  allowedTags: ['p', 'br', 'b', 'i', 'em', 'strong', 'a', 'ul', 'ol', 'li'],
  allowedAttributes: {
    'a': ['href']
  },
  allowedSchemes: ['http', 'https'],

const sanitizeInput = (input) => {

const db = require('../config/database');
const { sanitizeInput } = require('../utils/sanitize');

const getPosts = (req, res) => {
  db.all('SELECT id, title, content, created_at FROM posts ORDER BY created_at DESC', [], (err, rows) => {
    if (err) return res.status(500).json({ error: 'Database error' });
    res.json(rows);
  });
};

const getPost = (req, res) => {
  const { id } = req.params;
  db.get('SELECT * FROM posts WHERE id = ?', [id], (err, post) => {
    if (err) return res.status(500).json({ error: 'Database error' });
    if (!post) return res.status(404).json({ error: 'Post not found' });
    res.json(post);
  });
};

const createPost = (req, res) => {
  const { title, content } = req.body;
  
  const safeTitle = sanitizeInput(title);
  const safeContent = sanitizeInput(content);

  db.run(
    'INSERT INTO posts (title, content) VALUES (?, ?)',
    [safeTitle, safeContent],
    function(err) {
      if (err) return res.status(500).json({ error: 'Failed to create post' });
      res.status(201).json({ id: this.lastID, title: safeTitle, content: safeContent });
    }
  );
};

module.exports = { getPosts, getPost, createPost };

#### `frontend/js/utils.js`

```js
// Critical: DOM XSS Prevention
function escapeHtml(unsafe) {
  return unsafe
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#039;");
}

function formatDate(dateString) {
  return new Date(dateString).toLocaleDateString('en-US', {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
    hour: '2-digit',
    minute: '2-digit'
  });
}

// Client-side validation
function validateComment(author, content) {
  if (!author || author.trim().length < 2) {
    throw new Error('Author name must be at least 2 characters');
  }
  if (!content || content.trim().length < 3) {
    throw new Error('Comment must be at least 3 characters');
  }
  if (content.length > 2000) {
    throw new Error('Comment is too long (max 2000 characters)');
  }
  return true;
}
```

import api from './api.js';
import { escapeHtml, formatDate, validateComment } from './utils.js';

let currentPostId = null;

export async function loadComments(postId) {
  currentPostId = postId;
  const comments = await api.getComments(postId);
  const container = document.getElementById('comments-container');
  
  if (comments.length === 0) {
    container.innerHTML = '<p>No comments yet. Be the first!</p>';
    return;
  }

  container.innerHTML = comments.map(comment => `
    <div class="comment">
      <strong>${escapeHtml(comment.author_name)}</strong>
      <small>${formatDate(comment.created_at)}</small>
      <p>${escapeHtml(comment.content)}</p>
    </div>
  `).join('');
}

export function setupCommentForm() {
  const form = document.getElementById('comment-form');
  if (!form) return;

  form.addEventListener('submit', async (e) => {
    e.preventDefault();
    
    const author = document.getElementById('author').value;
    const content = document.getElementById('comment-content').value;

    try {
      validateComment(author, content);
      
      await api.addComment(currentPostId, { author_name: author, content });
      
      document.getElementById('author').value = '';
      document.getElementById('comment-content').value = '';
      
      await loadComments(currentPostId);
      showMessage('Comment posted successfully!', 'success');
    } catch (err) {
      showMessage(err.message, 'error');
    }
  });
}

function showMessage(message, type) {
  const msg = document.createElement('div');
  msg.className = type;
  msg.textContent = message;
  document.body.appendChild(msg);
  setTimeout(() => msg.remove(), 3000);
}

### Security Implementation Summary

**1. XSS Prevention (Critical):**
- **Backend**: `sanitize-html` with strict whitelist
- **Frontend**: `escapeHtml()` before rendering using `textContent` or manual escaping
- Never use `innerHTML` with unsanitized user content

**2. SQL Injection:**
- All queries use **parameterized queries** (SQLite `?` placeholders)

**3. Input Validation:**
- Backend: `express-validator` + `sanitize-html`
- Frontend: Client-side validation (not trusted)

**4. Rate Limiting:**
- 10 comments per 15 minutes per IP

**5. Other Protections:**
- Helmet.js security headers
- CSP (Content Security Policy)
- Payload size limits (100kb)
- Input length restrictions
- Proper error handling
