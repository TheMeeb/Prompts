### 3. `js/markdown-parser.js` (Core Security Layer)

```js
// js/markdown-parser.js
export class SecureMarkdownParser {
  constructor() {
    if (!window.marked || !window.DOMPurify) {
      throw new Error('Marked and DOMPurify must be loaded');
    }

    // Configure marked to be secure by default
    this.configureMarked();
  }

  configureMarked() {
    marked.setOptions({
      headerIds: false,
      mangle: false,
      breaks: true,
      gfm: true,
      // Critical: Disable raw HTML by default
      html: false,
    });
  }

  /**
   * Parse and sanitize markdown
   * @param {string} markdown
   * @returns {string} sanitized HTML
   */
  parse(markdown) {
    if (!markdown || typeof markdown !== 'string') {
      return '';
    }

    try {
      // 1. Convert Markdown to HTML (without allowing raw HTML)
      const rawHtml = marked.parse(markdown);

      // 2. Sanitize with DOMPurify (defense in depth)
      const cleanHtml = DOMPurify.sanitize(rawHtml, {
        ALLOWED_TAGS: [
          'h1', 'h2', 'h3', 'h4', 'h5', 'h6',
          'p', 'a', 'ul', 'ol', 'li', 'code', 'pre',
          'strong', 'em', 'blockquote', 'del', 'img',
          'table', 'thead', 'tbody', 'tr', 'th', 'td',
          'hr', 'br'
        ],
        ALLOWED_ATTR: [
          'href', 'src', 'alt', 'title', 'class',
          'target', 'rel'
        ],
        FORBID_TAGS: ['script', 'style', 'iframe', 'object', 'embed', 'form'],
        FORBID_ATTR: ['onerror', 'onload', 'onclick', 'onmouseover', 'style'],
        ADD_ATTR: { 'a': ['target', 'rel'] },
        // Prevent javascript: and data: URLs (except safe image data URIs)
        ALLOW_DATA_ATTR: false,
        SANITIZE_DOM: true,
        RETURN_DOM: false,
      });

      return cleanHtml || '';
    } catch (err) {
      console.error('Markdown parsing error:', err);
      return '<p style="color:red">Error rendering markdown.</p>';
    }
  }

  // For advanced use: allow specific safe HTML with explicit consent (not used by default)
  parseWithRawHtml(markdown) {
    // Only use if user explicitly enables "Allow HTML" (not recommended)
    marked.setOptions({ html: true });
    const html = marked.parse(markdown);
    marked.setOptions({ html: false }); // reset
    return DOMPurify.sanitize(html);
  }
}

// Singleton
export const parser = new SecureMarkdownParser();
```

// js/app.js
import { parser } from './markdown-parser.js';

class SecureMarkdownEditor {
  constructor() {
    this.input = document.getElementById('markdownInput');
    this.preview = document.getElementById('preview');
    this.debounceTimer = null;

    this.initEventListeners();
    this.loadSampleContent();
  }

  initEventListeners() {
    // Live preview with debounce
    this.input.addEventListener('input', () => {
      clearTimeout(this.debounceTimer);
      this.debounceTimer = setTimeout(() => this.renderPreview(), 80);
    });

    document.getElementById('clearBtn').addEventListener('click', () => {
      if (confirm('Clear editor?')) {
        this.input.value = '';
        this.renderPreview();
        this.input.focus();
      }
    });

    document.getElementById('saveBtn').addEventListener('click', () => this.saveDocument());
    document.getElementById('loadBtn').addEventListener('click', () => this.showSavedDocuments());
    document.querySelector('.close-modal').addEventListener('click', () => this.closeModal());
  }

  renderPreview() {
    const markdown = this.input.value;
    const html = parser.parse(markdown);
    this.preview.innerHTML = html;
  }

  loadSampleContent() {
    const sample = `# Secure Markdown Editor

Welcome! This editor is **protected** against XSS.

## Features
- Real-time preview
- Full XSS protection via **DOMPurify**
- Content Security Policy (CSP)
- Safe markdown parsing

\`\`\`js
console.log("This is safe");
\`\`\`

Try injecting: <script>alert('XSS')</script>
It **won't work**.
`;
    this.input.value = sample;
    this.renderPreview();
  }

  async saveDocument() {
    const title = prompt('Document title:', 'Untitled Document');
    if (!title) return;

    try {
      const res = await fetch('/api/documents', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          title: title.trim(),
          content: this.input.value
        })
      });

      const data = await res.json();
      if (data.success) alert('Document saved!');
    } catch (err) {
      console.error(err);
      alert('Failed to save document');
    }
  }

  async showSavedDocuments() {
    try {
      const res = await fetch('/api/documents');
      const docs = await res.json();
      
      const list = document.getElementById('documentList');
      list.innerHTML = docs.map(doc => `
        <div class="doc-item" onclick="window.loadDocument(${doc.id})">
          <strong>${this.escapeHtml(doc.title)}</strong>
          <small>${new Date(doc.updated_at).toLocaleString()}</small>
        </div>
      `).join('');

      document.getElementById('modal').classList.remove('hidden');
    } catch (e) {
      console.error(e);
    }
  }

  closeModal() {
    document.getElementById('modal').classList.add('hidden');
  }

  escapeHtml(unsafe) {
    return unsafe
      .replace(/&/g, "&amp;")
      .replace(/</g, "&lt;")
      .replace(/>/g, "&gt;");
  }
}

// Make loadDocument available globally for inline onclick
window.loadDocument = async function(id) {
  const res = await fetch(`/api/documents/${id}`);
  const doc = await res.json();
  document.getElementById('markdownInput').value = doc.content;
  document.getElementById('modal').classList.add('hidden');
  const event = new Event('input');
  document.getElementById('markdownInput').dispatchEvent(event);
};

// Initialize
document.addEventListener('DOMContentLoaded', () => {
  new SecureMarkdownEditor();
});

### 5. Backend (`server/server.js`)

```js
// server/server.js
import express from 'express';
import cors from 'cors';
import helmet from 'helmet';
import { initDb, saveDocument, getDocuments, getDocument } from './db.js';

const app = express();

app.use(helmet({
  contentSecurityPolicy: {
    directives: {
      defaultSrc: ["'self'"],
      scriptSrc: ["'self'"],
      styleSrc: ["'self'", "'unsafe-inline'"],
      imgSrc: ["'self'", "data:", "https:"],
    }
  }
}));

app.use(cors());
app.use(express.json());
app.use(express.static('.'));

app.post('/api/documents', async (req, res) => {
  const { title, content } = req.body;
  if (!title || !content) return res.status(400).json({ error: 'Missing fields' });

  try {
    const id = await saveDocument(title, content);
    res.json({ success: true, id });
  } catch (err) {
    res.status(500).json({ error: 'Server error' });
  }
});

app.get('/api/documents', async (req, res) => {
  const docs = await getDocuments();
  res.json(docs);
});

app.get('/api/documents/:id', async (req, res) => {
  const doc = await getDocument(req.params.id);
  if (!doc) return res.status(404).json({ error: 'Not found' });
  res.json(doc);
});

async function start() {
  await initDb();
  app.listen(3000, () => {
    console.log('Secure Markdown Editor running at http://localhost:3000');
    console.log('XSS Protection: Enabled (DOMPurify + CSP + HTML disabled)');
  });
}

start();
```

### Security Decisions (Critical)

| Measure | Implementation | Purpose |
|-------|----------------|--------|
| **Raw HTML Disabled** | `marked({ html: false })` | Prevents `<script>`, `<iframe>`, etc. |
| **DOMPurify Sanitization** | Deep config with allowlists | Defense-in-depth even if marked is bypassed |
| **CSP Headers** | Helmet + meta tag | Blocks inline scripts/styles |
| **No `innerHTML` without sanitization** | All output goes through parser | Prevents DOM-based XSS |
| **Input Validation** | Type + length checks | Backend validation |
| **Forbidden Tags/Attrs** | Explicit blacklist in DOMPurify | Catches `onerror`, `javascript:`, etc. |
