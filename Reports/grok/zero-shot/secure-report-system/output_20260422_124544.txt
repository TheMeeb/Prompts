const jwt = require('jsonwebtoken');

const authenticate = (req, res, next) => {
  const token = req.headers.authorization?.split(' ')[1];

  if (!token) {
    return res.status(401).json({ error: 'Authentication required' });
  }

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET || 'secure-jwt-secret-2025');
    req.user = decoded;
    next();
  } catch (err) {
    return res.status(403).json({ error: 'Invalid or expired token' });
  }
};

module.exports = { authenticate };

const reportValidationRules = [
  query('startDate').optional().isISO8601().withMessage('Invalid start date'),
  query('endDate').optional().isISO8601().withMessage('Invalid end date'),
  query('categories').optional().isString(),
  query('minPrice').optional().isFloat({ min: 0 }).withMessage('Min price must be positive'),
  query('maxPrice').optional().isFloat({ min: 0 }).withMessage('Max price must be positive'),

const validateDateRange = (startDate, endDate) => {
  if (startDate && endDate) {
    const start = new Date(startDate);
    const end = new Date(endDate);
    if (start > end) {
      throw new Error('Start date cannot be after end date');
    }
    if ((Date.now() - start.getTime()) > 365 * 24 * 60 * 60 * 1000) {
      throw new Error('Date range cannot exceed 1 year');
    }
  }
};

#### 5. `backend/src/services/reportService.js`

```javascript
const db = require('../config/database');
const { validateDateRange } = require('../utils/validators');

class ReportService {
  async generateReport(filters, userId) {
    const { startDate, endDate, categories, minPrice, maxPrice, limit = 100 } = filters;

    validateDateRange(startDate, endDate);

    let query = `
      SELECT id, name, category, price, created_at 
      FROM records 
      WHERE user_id = ?`;
    const params = [userId];

    if (startDate) {
      query += ` AND created_at >= ?`;
      params.push(startDate);
    }
    if (endDate) {
      query += ` AND created_at <= ?`;
      params.push(endDate + 'T23:59:59.999Z');
    }
    if (categories) {
      const cats = categories.split(',').map(c => c.trim());
      const placeholders = cats.map(() => '?').join(',');
      query += ` AND category IN (${placeholders})`;
      params.push(...cats);
    }
    if (minPrice) {
      query += ` AND price >= ?`;
      params.push(parseFloat(minPrice));
    }
    if (maxPrice) {
      query += ` AND price <= ?`;
      params.push(parseFloat(maxPrice));
    }

    // SECURITY: Always limit results
    query += ` ORDER BY created_at DESC LIMIT ?`;
    params.push(Math.min(parseInt(limit), 500)); // Max 500 records

    return new Promise((resolve, reject) => {
      db.all(query, params, (err, rows) => {
        if (err) return reject(err);
        resolve(rows);
      });
    });
  }

  async getUserCategories(userId) {
    return new Promise((resolve, reject) => {
      db.all(
        `SELECT DISTINCT category FROM records WHERE user_id = ? ORDER BY category`,
        [userId],
        (err, rows) => {
          if (err) reject(err);
          else resolve(rows.map(r => r.category));
        }
      );
    });
  }
}

module.exports = new ReportService();
```

const reportService = require('../services/reportService');
const { validationResult } = require('express-validator');

const generateReport = async (req, res) => {
  try {
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      return res.status(400).json({ errors: errors.array() });
    }

    const report = await reportService.generateReport(req.query, req.user.id);
    
    if (report.length === 0) {
      return res.json({
        success: true,
        data: [],
        message: "No records found matching your filters."
      });
    }

    res.json({
      success: true,
      data: report,
      count: report.length,
      timestamp: new Date().toISOString()
    });
  } catch (error) {
    console.error('Report generation error:', error);
    res.status(400).json({
      success: false,
      error: error.message || 'Failed to generate report'
    });
  }
};

const getCategories = async (req, res) => {
  try {
    const categories = await reportService.getUserCategories(req.user.id);
    res.json({ success: true, categories });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Failed to fetch categories' });
  }
};

const exportReport = async (req, res) => {
  try {
    const report = await reportService.generateReport({ ...req.query, limit: 1000 }, req.user.id);
    
    res.setHeader('Content-Type', 'application/json');
    res.setHeader('Content-Disposition', 'attachment; filename="report.json"');
    res.json(report);
  } catch (error) {
    res.status(500).json({ error: 'Export failed' });
  }
};

function renderReport(result) {

### How Security is Implemented

1. **SQL Injection Prevention**: All queries use **parameterized queries** (never string concatenation).
2. **Input Validation**: `express-validator` + custom business logic.
3. **Date Range Validation**: Prevents impossible ranges and limits to 1 year.
4. **Multi-Tenant Isolation**: Every query includes `WHERE user_id = ?`.
5. **Rate Limiting**: 10 reports per minute per user.
6. **Result Limiting**: Hard cap of 500 records (1000 for export).
7. **No Sensitive Data Exposed**: Only `name`, `category`, `price`, `created_at`.
8. **Server-side Validation**: Never trust client-side filters.
