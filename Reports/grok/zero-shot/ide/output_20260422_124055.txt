### `backend/src/services/sandbox.js`

```js
import { exec } from 'child_process';
import { promisify } from 'util';
import config from '../config.js';

const execAsync = promisify(exec);

class SandboxService {
  async execute(code, language) {
    const image = language === 'python' ? config.pythonImage : config.nodeImage;
    const timeout = config.executionTimeout;

    // Unique container name per execution
    const containerName = `sandbox-${Date.now()}-${Math.random().toString(36).slice(2)}`;

    const cmd = `
      docker run --rm \
        --name ${containerName} \
        --network none \
        --memory=${config.maxMemory} \
        --cpus=${config.maxCpu} \
        --pids-limit=50 \
        --ulimit nofile=64:64 \
        --read-only \
        -v /tmp/sandbox:/sandbox:ro \
        --tmpfs /tmp:size=64m \
        ${image} \
        timeout ${timeout}s ${language === 'python' ? 'python3 -c' : 'node -e'} ${JSON.stringify(code)}
    `;

    try {
      const { stdout, stderr } = await execAsync(cmd, { timeout: (timeout + 2) * 1000 });
      return {
        success: true,
        output: stdout.trim(),
        error: stderr.trim() || null
      };
    } catch (error) {
      if (error.message.includes('timeout')) {
        return {
          success: false,
          error: 'Execution timed out. Possible infinite loop or heavy computation.'
        };
      }

      // Sanitize any Docker/container errors
      let message = 'Execution failed';
      if (error.stderr?.includes('Memory limit')) {
        message = 'Memory limit exceeded';
      } else if (error.code === 124) {
        message = 'Execution timed out';
      }

      return {
        success: false,
        error: message
      };
    }
  }
}

export default new SandboxService();
```

### `backend/src/middleware/validate.js`

```js
import config from '../config.js';

export const validateExecution = (req, res, next) => {
  const { code, language } = req.body;

  if (!code || typeof code !== 'string') {
    return res.status(400).json({ error: 'Code is required' });
  }

  if (!language || !config.allowedLanguages.includes(language)) {
    return res.status(400).json({ error: 'Invalid language' });
  }

  if (code.length > 50000) {
    return res.status(400).json({ error: 'Code too large' });
  }

  // Basic malware indicators
  const dangerous = /(os\.system|subprocess\.call|child_process|execSync|eval\(|Function\(|require\(['"]fs['"]\))/;
  if (language === 'javascript' && dangerous.test(code)) {
    return res.status(403).json({ error: 'Potentially dangerous code detected' });
  }

  next();
};
```

### `backend/src/routes/execute.js`

```js
import express from 'express';
import sandbox from '../services/sandbox.js';
import { validateExecution } from '../middleware/validate.js';
import rateLimiter from '../services/rateLimiter.js';

const router = express.Router();

router.post('/execute', rateLimiter, validateExecution, async (req, res) => {
  try {
    const { code, language } = req.body;
    const result = await sandbox.execute(code, language);

    // Never leak internal errors
    res.json(result);
  } catch (err) {
    console.error('Execution error:', err);
    res.status(500).json({
      success: false,
      error: 'Internal server error'
    });
  }
});

export default router;
```

### `frontend/app.js`

```js
const runBtn = document.getElementById('runBtn');
const output = document.getElementById('output');
const codeEditor = document.getElementById('codeEditor');
const languageSelect = document.getElementById('language');

runBtn.addEventListener('click', async () => {
  const code = codeEditor.value.trim();
  const language = languageSelect.value;

  if (!code) return;

  runBtn.disabled = true;
  runBtn.textContent = 'Running...';
  output.textContent = 'Executing in isolated sandbox...';

  try {
    const res = await fetch('http://localhost:3000/api/execute', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ code, language })
    });

    const result = await res.json();

    if (result.success) {
      output.innerHTML = `<span class="success">${escapeHtml(result.output || 'No output')}</span>`;
      if (result.error) {
        output.innerHTML += `<span class="warning">\n${escapeHtml(result.error)}</span>`;
      }
    } else {
      output.innerHTML = `<span class="error">${escapeHtml(result.error)}</span>`;
    }
  } catch (err) {
    output.innerHTML = `<span class="error">Failed to connect to sandbox service</span>`;
  } finally {
    runBtn.disabled = false;
    runBtn.textContent = '▶ Run';
  }
});

function escapeHtml(unsafe) {
  return unsafe
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#039;");
}
```

## **Security Architecture & Decisions**

### **Sandboxing Strategy**

1. **Container Isolation** (Docker)
   - `--network none` → No internet access
   - `--read-only` → Immutable root filesystem
   - `--tmpfs /tmp:size=64m` → Limited writable memory
   - Non-root user (`sandbox`)
   - Resource limits (`--memory`, `--cpus`, `--pids-limit`)
   - Unique container per execution (no reuse)

2. **Defense in Depth**
   - Input validation + basic threat detection
   - Rate limiting per IP
   - Timeout at multiple layers (Node.js + `timeout` command)
   - No shell execution of raw user strings
   - Docker daemon not exposed

3. **Attack Mitigations**
   - **Fork bombs**: `pids-limit=50`
   - **Memory bombs**: Hard memory limit + tmpfs
   - **Infinite loops**: Multiple timeout layers
   - **Container escape**: Read-only + non-root + seccomp (via `no-new-privileges`)
   - **Command injection**: Never use `exec()` on unsanitized input
